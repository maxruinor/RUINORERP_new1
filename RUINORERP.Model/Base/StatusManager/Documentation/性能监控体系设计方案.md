# RuinorERP 系统性能监控体系设计方案

## 1. 监控范围确定

### 1.1 客户端性能监控
- **本地执行时间**：方法执行耗时统计
- **UI响应时间**：窗体加载、控件渲染时间
- **内存使用**：客户端内存占用监控
- **线程状态**：线程池使用情况

### 1.2 网络性能监控
- **请求响应时间**：SuperSocket 请求-响应时间
- **数据传输速度**：数据包传输速率
- **连接状态**：连接稳定性、重连次数
- **网络延迟**：往返延迟统计

### 1.3 数据库性能监控
- **查询执行时间**：SQL 查询耗时统计
- **事务执行时间**：事务开始-提交时间
- **连接池状态**：数据库连接使用情况
- **死锁检测**：死锁发生频率和持续时间

## 2. 监控数据呈现形式

### 2.1 日志系统设计

#### 日志格式
```
[时间] [级别] [类别] [方法] 消息
[2026-04-03 10:30:45.123] [INFO] [Performance] [GetOrCreate] 缓存命中，耗时：2.5ms
[2026-04-03 10:30:45.456] [WARN] [Database] [ExecuteQuery] 查询超时，耗时：5000ms
```

#### 日志级别
- **Trace**：详细执行跟踪
- **Debug**：调试信息
- **Information**：一般信息
- **Warning**：警告信息
- **Error**：错误信息
- **Critical**：严重错误

#### 存储策略
- **文件日志**：按日期滚动存储
- **数据库日志**：重要性能数据持久化
- **内存日志**：实时性能数据

### 2.2 监控UI设计

#### 管理员视图
- **实时性能仪表板**：显示当前系统性能指标
- **历史趋势图表**：展示性能变化趋势
- **异常统计报告**：统计异常发生频率和类型
- **告警通知中心**：集中显示所有告警信息

#### 开发者视图
- **方法执行时间线**：显示方法调用耗时
- **调用链追踪**：追踪方法调用关系
- **内存使用曲线**：显示内存占用变化

```csharp
    private readonly ConcurrentDictionary<string, ConnectionMetrics> _connectionMetrics;
    private readonly ConcurrentQueue<RequestMetrics> _requestHistory;
    
    /// <summary>
    /// 连接指标
    /// </summary>
    public class ConnectionMetrics
    {
        public string ConnectionId { get; set; }
        public DateTime ConnectTime { get; set; }
        public int ReconnectCount { get; set; }
        public long TotalBytesSent { get; set; }
        public long TotalBytesReceived { get; set; }
        public double AverageLatencyMs { get; set; }
    }
    
    /// <summary>
    /// 请求指标
    /// </summary>
    public class RequestMetrics
    {
        public string RequestId { get; set; }
        public DateTime StartTime { get; set; }
        public DateTime EndTime { get; set; }
        public long DurationMs { get; set; }
        public int BytesSent { get; set; }
        public int BytesReceived { get; set; }
        public bool Success { get; set; }
    }
    
    /// <summary>
    /// 记录请求开始
    /// </summary>
    public string RecordRequestStart(string requestId, int bytesSent)
    {
        var metrics = new RequestMetrics
        {
            RequestId = requestId,
            StartTime = DateTime.Now,
            BytesSent = bytesSent
        };
        _requestHistory.Enqueue(metrics);
        return requestId;
    }
    
    /// <summary>
    /// 记录请求结束
    /// </summary>
    public void RecordRequestEnd(string requestId, int bytesReceived, bool success)
    {
        var metrics = _requestHistory.FirstOrDefault(m => m.RequestId == requestId);
        if (metrics != null)
        {
            metrics.EndTime = DateTime.Now;
            metrics.DurationMs = (long)(metrics.EndTime - metrics.StartTime).TotalMilliseconds;
            metrics.BytesReceived = bytesReceived;
            metrics.Success = success;
        }
    }
}
```

```csharp
/// <summary>
/// 批量性能数据传输器
/// </summary>
public class BatchPerformanceTransmitter
{
    private readonly ISuperSocketClient _socketClient;
    private readonly ILogger _logger;
    private readonly ConcurrentQueue<PerformanceData> _dataQueue;
    private readonly Timer _batchTimer;
    private const int BatchSize = 100;
    private const int BatchIntervalMs = 5000; // 5秒
    
    /// <summary>
    /// 添加性能数据
    /// </summary>
    public void AddPerformanceData(PerformanceData data)
    {
        _dataQueue.Enqueue(data);
        
        if (_dataQueue.Count >= BatchSize)
        {
            FlushBatch();
        }
    }
    
    /// <summary>
    /// 刷新批量数据
    /// </summary>
    private void FlushBatch()
    {
        var batch = new List<PerformanceData>();
        while (_dataQueue.TryDequeue(out var data) && batch.Count < BatchSize)
        {
            batch.Add(data);
        }
        
        if (batch.Count > 0)
        {
            SendBatchAsync(batch);
        }
    }
}
```

```csharp
        _realtimeData[data.DataId] = data;

    /// 获取最近数据
```

```csharp
/// <summary>
/// 实时性能分析器
/// </summary>
public class RealtimePerformanceAnalyzer
{
    private readonly ILogger _logger;
    private readonly Dictionary<string, PerformanceThreshold> _thresholds;
    
    /// <summary>
    /// 性能阈值
    /// </summary>
    public class PerformanceThreshold
    {
        public string MetricName { get; set; }
        public double WarningThreshold { get; set; }
        public double CriticalThreshold { get; set; }
        public TimeSpan CheckInterval { get; set; }
    }
    
    /// <summary>
    /// 分析性能数据
    /// </summary>
    public List<PerformanceAlert> AnalyzePerformanceData(PerformanceData data)
    {
        var alerts = new List<PerformanceAlert>();
        
        foreach (var metric in data.Metrics)
        {
            var threshold = _thresholds.GetValueOrDefault(metric.MetricType);
            if (threshold != null)
            {
                if (metric.Value >= threshold.CriticalThreshold)
                {
                    alerts.Add(new PerformanceAlert
                    {
                        AlertType = AlertType.Critical,
                        MetricType = metric.MetricType,
                        Message = $"{metric.MetricType} 超过临界阈值：{metric.Value}",
                        Severity = AlertSeverity.Critical,
                        Timestamp = DateTime.Now
                    });
                }
                else if (metric.Value >= threshold.WarningThreshold)
                {
                    alerts.Add(new PerformanceAlert
                    {
                        AlertType = AlertType.Warning,
                        MetricType = metric.MetricType,
                        Message = $"{metric.MetricType} 超过警告阈值：{metric.Value}",
                        Severity = AlertSeverity.Warning,
                        Timestamp = DateTime.Now
                    });
                }
            }
        }
        
        return alerts;
    }
}
```

```csharp
/// <summary>

public class HistoricalPerformanceAnalyzer
{
    private readonly ILogger _logger;
    
    /// <summary>
    /// 性能趋势
    /// </summary>
    public class PerformanceTrend
    {
        public string MetricType { get; set; }
        public double AverageValue { get; set; }
        public double MaxValue { get; set; }
        public double MinValue { get; set; }
        public TrendDirection Direction { get; set; }
        public double ChangeRate { get; set; }
    }
    
    /// <summary>
    /// 趋势方向
    /// </summary>
    public enum TrendDirection
    {
        Increasing,
        Decreasing,
        Stable,
        Unknown
    }
    
    /// <summary>
    /// 分析性能趋势
    /// </summary>
    public List<PerformanceTrend> AnalyzeTrends(List<PerformanceMetric> metrics, string metricType, int days = 7)
    {
        var trends = new List<PerformanceTrend>();
        var groupedMetrics = metrics
            .Where(m => m.MetricType == metricType)
            .GroupBy(m => m.Timestamp.Date)
            .ToList();
        
        foreach (var group in groupedMetrics)
        {
            var trend = new PerformanceTrend
            {
                MetricType = metricType,
                AverageValue = group.Average(m => m.MetricValue),
                MaxValue = group.Max(m => m.MetricValue),
                MinValue = group.Min(m => m.MetricValue)
            };
            
            trends.Add(trend);
        }
        
        return trends;
    }
}
```

##### 3. 异常检测
```csharp
/// <summary>
/// 异常检测器
/// </summary>
public class AnomalyDetector
{
    private readonly ILogger _logger;
    
    /// <summary>
    /// 检测死锁
    /// </summary>
    public List<PerformanceAlert> DetectDeadlocks(List<DatabaseMetric> metrics)
    {
        var alerts = new List<PerformanceAlert>();
        
        var longRunningQueries = metrics
            .Where(m => m.ExecutionTimeMs > 30000) // 超过30秒
            .ToList();
        
        foreach (var query in longRunningQueries)
        {
            alerts.Add(new PerformanceAlert
            {
                AlertType = AlertType.Database,
                MetricType = "Deadlock",
                Message = $"检测到可能的死锁，查询执行时间：{query.ExecutionTimeMs}ms",
                Severity = AlertSeverity.Critical,
                Timestamp = DateTime.Now
            });
        }
        
        return alerts;
    }
    
    /// <summary>
    /// 检测内存泄漏
    /// </summary>
    public List<PerformanceAlert> DetectMemoryLeaks(List<PerformanceMetric> metrics)
    {
        var alerts = new List<PerformanceAlert>();
        
        var memoryMetrics = metrics
            .Where(m => m.MetricType == "MemoryUsage")
            .OrderBy(m => m.Timestamp)
            .ToList();
        
        for (int i = 1; i < memoryMetrics.Count; i++)
        {
            var current = memoryMetrics[i];
            var previous = memoryMetrics[i - 1];
            
            if (current.MetricValue > previous.MetricValue * 1.5) // 增长和其他50%
            {
                alerts.Add(new PerformanceAlert
                {
                    AlertType = AlertType.Memory,
                    MetricType = "MemoryLeak",
                    Message = $"检测到可能的内存泄漏，内存使用增长：{current.MetricValue}MB",
                    Severity = AlertSeverity.Warning,
                    Timestamp = DateTime.Now
                });
            }
        }
        
        return alerts;
    }
}
```

```csharp
/// <summary>
/// 手动同步器
/// </summary>
public class ManualSynchronizer
{
    private readonly ISuperSocketClient _socketClient;
    private readonly ILogger _logger;
    
    /// <summary>
    /// 手动同步所有数据
    /// </summary>
    public async Task SyncAllDataAsync(IProgress<SyncProgress> progress)
    {
        try
        {
            var data = await LoadAllLocalPerformanceDataAsync();
            var total = data.Count;
            var current = 0;
            
            foreach (var batch in data.Batch(100))
            {
                var packet = new PerformanceDataPacket
                {
                    PacketType = PacketType.PerformanceData,
                    ClientID = Environment.MachineName,
                    Timestamp = DateTime.Now,
                    Metrics = batch.ToList()
                };
                
                await _socketClient.SendPacketAsync(packet);
                
                current += batch.Count();
                progress?.Report(new SyncProgress
                {
                    Current = current,
                    Total = total,
                    Percentage = (double)current / total * 100
                });
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "手动同步失败");
        }
    }
}
```
