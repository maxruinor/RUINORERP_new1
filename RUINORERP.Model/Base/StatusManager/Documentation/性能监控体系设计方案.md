# RuinorERP 系统性能监控体系设计方案

## 1. 监控范围确定

### 1.1 客户端性能监控
- **本地执行时间**：方法执行耗时统计
- **UI响应时间**：窗体加载、控件渲染时间
- **内存使用**：客户端内存占用监控
- **线程状态**：线程池使用情况

### 1.2 网络性能监控
- **请求响应时间**：SuperSocket 请求-响应时间
- **数据传输速度**：数据包传输速率
- **连接状态**：连接稳定性、重连次数
- **网络延迟**：往返延迟统计

### 1.3 数据库性能监控
- **查询执行时间**：SQL 查询耗时统计
- **事务执行时间**：事务开始-提交时间
- **连接池状态**：数据库连接使用情况
- **死锁检测**：死锁发生频率和持续时间

## 2. 监控数据呈现形式

### 2.1 日志系统设计

#### 日志格式
```
[时间] [级别] [类别] [方法] 消息
[2026-04-03 10:30:45.123] [INFO] [Performance] [GetOrCreate] 缓存命中，耗时：2.5ms
[2026-04-03 10:30:45.456] [WARN] [Database] [ExecuteQuery] 查询超时，耗时：5000ms
```

#### 日志级别
- **Trace**：详细执行跟踪
- **Debug**：调试信息
- **Information**：一般信息
- **Warning**：警告信息
- **Error**：错误信息
- **Critical**：严重错误

#### 存储策略
- **文件日志**：按日期滚动存储
- **数据库日志**：重要性能数据持久化
- **内存日志**：实时性能数据

### 2.2 监控UI设计

#### 管理员视图
- **实时性能仪表板**：显示当前系统性能指标
- **历史趋势图表**：展示性能变化趋势
- **异常统计报告**：统计异常发生频率和类型
- **告警通知中心**：集中显示所有告警信息

#### 开发者视图
- **方法执行时间线**：显示方法调用耗时
- **调用链追踪**：追踪方法调用关系
- **内存使用曲线**：显示内存占用变化

```csharp
    private readonly ConcurrentDictionary<string, ConnectionMetrics> _connectionMetrics;
    private readonly ConcurrentQueue<RequestMetrics> _requestHistory;
    
    /// <summary>
    /// 连接指标
    /// </summary>
    public class ConnectionMetrics
    {
        public string ConnectionId { get; set; }
        public DateTime ConnectTime { get; set; }
        public int ReconnectCount { get; set; }
        public long TotalBytesSent { get; set; }
        public long TotalBytesReceived { get; set; }
        public double AverageLatencyMs { get; set; }
    }
    
    /// <summary>
    /// 请求指标
    /// </summary>
    public class RequestMetrics
    {
        public string RequestId { get; set; }
        public DateTime StartTime { get; set; }
        public DateTime EndTime { get; set; }
        public long DurationMs { get; set; }
        public int BytesSent { get; set; }
        public int BytesReceived { get; set; }
        public bool Success { get; set; }
    }
    
    /// <summary>
    /// 记录请求开始
    /// </summary>
    public string RecordRequestStart(string requestId, int bytesSent)
    {
        var metrics = new RequestMetrics
        {
            RequestId = requestId,
            StartTime = DateTime.Now,
            BytesSent = bytesSent
        };
        _requestHistory.Enqueue(metrics);
        return requestId;
    }
    
    /// <summary>
    /// 记录请求结束
    /// </summary>
    public void RecordRequestEnd(string requestId, int bytesReceived, bool success)
    {
        var metrics = _requestHistory.FirstOrDefault(m => m.RequestId == requestId);
        if (metrics != null)
        {
            metrics.EndTime = DateTime.Now;
            metrics.DurationMs = (long)(metrics.EndTime - metrics.StartTime).TotalMilliseconds;
            metrics.BytesReceived = bytesReceived;
            metrics.Success = success;
        }
    }
}
```

```csharp
/// <summary>
/// 批量性能数据传输器
/// </summary>
public class BatchPerformanceTransmitter
{
    private readonly ISuperSocketClient _socketClient;
    private readonly ILogger _logger;
    private readonly ConcurrentQueue<PerformanceData> _dataQueue;
    private readonly Timer _batchTimer;
    private const int BatchSize = 100;
    private const int BatchIntervalMs = 5000; // 5秒
    
    /// <summary>
    /// 添加性能数据
    /// </summary>
    public void AddPerformanceData(PerformanceData data)
    {
        _dataQueue.Enqueue(data);
        
        if (_dataQueue.Count >= BatchSize)
        {
            FlushBatch();
        }
    }
    
    /// <summary>
    /// 刷新批量数据
    /// </summary>
    private void FlushBatch()
    {
        var batch = new List<PerformanceData>();
        while (_dataQueue.TryDequeue(out var data) && batch.Count < BatchSize)
        {
            batch.Add(data);
        }
        
        if (batch.Count > 0)
        {
            SendBatchAsync(batch);
        }
    }
}
```

```csharp
        _realtimeData[data.DataId] = data;

    /// 获取最近数据
```

```csharp
/// <summary>
/// 实时性能分析器
/// </summary>
public class RealtimePerformanceAnalyzer
{
    private readonly ILogger _logger;
    private readonly Dictionary<string, PerformanceThreshold> _thresholds;
    
    /// <summary>
    /// 性能阈值
    /// </summary>
    public class PerformanceThreshold
    {
        public string MetricName { get; set; }
        public double WarningThreshold { get; set; }
        public double CriticalThreshold { get; set; }
        public TimeSpan CheckInterval { get; set; }
    }
    
    /// <summary>
    /// 分析性能数据
    /// </summary>
    public List<PerformanceAlert> AnalyzePerformanceData(PerformanceData data)
    {
        var alerts = new List<PerformanceAlert>();
        
        foreach (var metric in data.Metrics)
        {
            var threshold = _thresholds.GetValueOrDefault(metric.MetricType);
            if (threshold != null)
            {
                if (metric.Value >= threshold.CriticalThreshold)
                {
                    alerts.Add(new PerformanceAlert
                    {
                        AlertType = AlertType.Critical,
                        MetricType = metric.MetricType,
                        Message = $"{metric.MetricType} 超过临界阈值：{metric.Value}",
                        Severity = AlertSeverity.Critical,
                        Timestamp = DateTime.Now
                    });
                }
                else if (metric.Value >= threshold.WarningThreshold)
                {
                    alerts.Add(new PerformanceAlert
                    {
                        AlertType = AlertType.Warning,
                        MetricType = metric.MetricType,
                        Message = $"{metric.MetricType} 超过警告阈值：{metric.Value}",
                        Severity = AlertSeverity.Warning,
                        Timestamp = DateTime.Now
                    });
                }
            }
        }
        
        return alerts;
    }
}
```

```csharp
/// <summary>

public class HistoricalPerformanceAnalyzer
{
    private readonly ILogger _logger;
    
    /// <summary>
    /// 性能趋势
    /// </summary>
    public class PerformanceTrend
    {
        public string MetricType { get; set; }
        public double AverageValue { get; set; }
        public double MaxValue { get; set; }
        public double MinValue { get; set; }
        public TrendDirection Direction { get; set; }
        public double ChangeRate { get; set; }
    }
    
    /// <summary>
    /// 趋势方向
    /// </summary>
    public enum TrendDirection
    {
        Increasing,
        Decreasing,
        Stable,
        Unknown
    }
    
    /// <summary>
    /// 分析性能趋势
    /// </summary>
    public List<PerformanceTrend> AnalyzeTrends(List<PerformanceMetric> metrics, string metricType, int days = 7)
    {
        var trends = new List<PerformanceTrend>();
        var groupedMetrics = metrics
            .Where(m => m.MetricType == metricType)
            .GroupBy(m => m.Timestamp.Date)
            .ToList();
        
        foreach (var group in groupedMetrics)
        {
            var trend = new PerformanceTrend
            {
                MetricType = metricType,
                AverageValue = group.Average(m => m.MetricValue),
                MaxValue = group.Max(m => m.MetricValue),
                MinValue = group.Min(m => m.MetricValue)
            };
            
            trends.Add(trend);
        }
        
        return trends;
    }
}
```

##### 3. 异常检测
```csharp
/// <summary>
/// 异常检测器
/// </summary>
public class AnomalyDetector
{
    private readonly ILogger _logger;
    
    /// <summary>
    /// 检测死锁
    /// </summary>
    public List<PerformanceAlert> DetectDeadlocks(List<DatabaseMetric> metrics)
    {
        var alerts = new List<PerformanceAlert>();
        
        var longRunningQueries = metrics
            .Where(m => m.ExecutionTimeMs > 30000) // 超过30秒
            .ToList();
        
        foreach (var query in longRunningQueries)
        {
            alerts.Add(new PerformanceAlert
            {
                AlertType = AlertType.Database,
                MetricType = "Deadlock",
                Message = $"检测到可能的死锁，查询执行时间：{query.ExecutionTimeMs}ms",
                Severity = AlertSeverity.Critical,
                Timestamp = DateTime.Now
            });
        }
        
        return alerts;
    }
    
    /// <summary>
    /// 检测内存泄漏
    /// </summary>
    public List<PerformanceAlert> DetectMemoryLeaks(List<PerformanceMetric> metrics)
    {
        var alerts = new List<PerformanceAlert>();
        
        var memoryMetrics = metrics
            .Where(m => m.MetricType == "MemoryUsage")
            .OrderBy(m => m.Timestamp)
            .ToList();
        
        for (int i = 1; i < memoryMetrics.Count; i++)
        {
            var current = memoryMetrics[i];
            var previous = memoryMetrics[i - 1];
            
            if (current.MetricValue > previous.MetricValue * 1.5) // 增长和其他50%
            {
                alerts.Add(new PerformanceAlert
                {
                    AlertType = AlertType.Memory,
                    MetricType = "MemoryLeak",
                    Message = $"检测到可能的内存泄漏，内存使用增长：{current.MetricValue}MB",
                    Severity = AlertSeverity.Warning,
                    Timestamp = DateTime.Now
                });
            }
        }
        
        return alerts;
    }
}
```

```csharp
/// <summary>
/// 告警升级器
/// </summary>
public class AlertEscalator
{
    private readonly ILogger _logger;
    private readonly Dictionary<string, AlertEscalationRule> _escalationRules;
    
    /// <summary>
    /// 告警升级规则
    /// </summary>
    public class AlertEscalationRule
    {
        public string AlertType { get; set; }
        public TimeSpan FirstEscalationTime { get; set; }
        public TimeSpan SecondEscalationTime { get; set; }
        public AlertSeverity FirstEscalationSeverity { get; set; }
        public AlertSeverity SecondEscalationSeverity { get; set; }
    }
    
    /// <summary>
    /// 检查是否需要升级告警
    /// </summary>
    public bool ShouldEscalate(PerformanceAlert alert)
    {
        var rule = _escalationRules.GetValueOrDefault(alert.AlertType);
        if (rule != null)
        {
            var timeSinceAlert = DateTime.Now - alert.Timestamp;
            
            if (timeSinceAlert >= rule.SecondEscalationTime)
            {
                return true;
            }
        }
        
        return false;
    }
    
    /// <summary>
    /// 升级告警
    /// </summary>
    public PerformanceAlert EscalateAlert(PerformanceAlert alert)
    {
        var rule = _escalationRules.GetValueOrDefault(alert.AlertType);
        if (rule != null)
        {
            var timeSinceAlert = DateTime.Now - alert.Timestamp;
            
            if (timeSinceAlert >= rule.FirstEscalationTime && timeSinceAlert < rule.SecondEscalationTime)
            {
                alert.Severity = rule.FirstEscalationSeverity;
            }
            else if (timeSinceAlert >= rule.SecondEscalationTime)
            {
                alert.Severity = rule.SecondEscalationSeverity;
            }
        }
        
        return alert;
    }
}
```

## 7. 核心组件详细设计

### 7.1 PerformanceMonitor 类详细设计

```csharp
using System;
using System.Collections.Concurrent;
using System.Diagnostics;
using System.Linq;
using Microsoft.Extensions.Logging;

namespace RUINORERP.Common.PerformanceMonitoring
{
    /// <summary>
    /// 性能监控器 - 核心监控组件
    /// </summary>
    public class PerformanceMonitor : IDisposable
    {
        private readonly ILogger _logger;
        private readonly ConcurrentDictionary<string, MethodMetrics> _methodMetrics;
        private readonly Timer _collectionTimer;
        private readonly object _lock = new object();
        private bool _disposed = false;
        
        /// <summary>
        /// 方法执行指标
        /// </summary>
        public class MethodMetrics
        {
            public string MethodName { get; set; }
            public long TotalCalls { get; set; }
            public long TotalTimeMs { get; set; }
            public long MaxTimeMs { get; set; }
            public long MinTimeMs { get; set; }
            public double AverageTimeMs => TotalCalls > 0 ? (double)TotalTimeMs / TotalCalls : 0;
        }
        
        /// <summary>
        /// 性能统计
        /// </summary>
        public class PerformanceStats
        {
            public List<MethodMetrics> MethodMetrics { get; set; }
            public long TotalMemory { get; set; }
            public long AvailableMemory { get; set; }
            public int ThreadCount { get; set; }
            public DateTime CollectTime { get; set; }
        }
        
        public PerformanceMonitor(ILogger<PerformanceMonitor> logger)
        {
            _logger = logger;
            _collectionTimer = new Timer(CollectStats, null, TimeSpan.FromSeconds(60), TimeSpan.FromSeconds(60));
        }
        
        /// <summary>
        /// 监控方法执行
        /// </summary>
        public IDisposable MonitorMethod(string methodName)
        {
            return new MethodPerformanceTracker(this, methodName);
        }
        
        /// <summary>
        /// 获取方法统计
        /// </summary>
        public MethodMetrics GetMethodStats(string methodName)
        {
            lock (_lock)
            {
                return _methodMetrics.GetValueOrDefault(methodName, new MethodMetrics { MethodName = methodName });
            }
        }
        
        /// <summary>
        /// 获取所有方法统计
        /// </summary>
        public List<MethodMetrics> GetAllMethodStats()
        {
            lock (_lock)
            {
                return _methodMetrics.Values.ToList();
            }
        }
        
        /// <summary>
        /// 获取性能统计
        /// </summary>
        public PerformanceStats GetStats()
        {
            lock (_lock)
            {
                return new PerformanceStats
                {
                    MethodMetrics = _methodMetrics.Values.ToList(),
                    TotalMemory = GC.GetTotalMemory(false),
                    AvailableMemory = GC.GetTotalMemory(true),
                    ThreadCount = ThreadPool.ThreadCount,
                    CollectTime = DateTime.Now
                };
            }
        }
        
        /// <summary>
        /// 重置统计
        /// </summary>
        public void ResetStats()
        {
            lock (_lock)
            {
                _methodMetrics.Clear();
            }
        }
        
        /// <summary>
        /// 收集统计信息
        /// </summary>
        private void CollectStats(object state)
        {
            try
            {
                var stats = GetStats();
                _logger.LogInformation("性能统计收集完成，方法数：{MethodCount}，总内存：{TotalMemory}MB，可用内存：{AvailableMemory}MB，线程数：{ThreadCount}",
                    stats.MethodMetrics.Count, stats.TotalMemory / 1024 / 1024, stats.AvailableMemory / 1024 / 1024, stats.ThreadCount);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "收集性能统计失败");
            }
        }
        
        /// <summary>
        /// 释放资源
        /// </summary>
        public void Dispose()
        {
            if (!_disposed)
            {
                _collectionTimer?.Dispose();
                _disposed = true;
            }
        }
    }
    
    /// <summary>
    /// 方法性能跟踪器
    /// </summary>
    public class MethodPerformanceTracker : IDisposable
    {
        private readonly PerformanceMonitor _monitor;
        private readonly string _methodName;
        private readonly Stopwatch _stopwatch;
        private bool _disposed = false;
        
        public MethodPerformanceTracker(PerformanceMonitor monitor, string methodName)
        {
            _monitor = monitor;
            _methodName = methodName;
            _stopwatch = Stopwatch.StartNew();
        }
        
        /// <summary>
        /// 释放资源
        /// </summary>
        public void Dispose()
        {
            if (!_disposed)
            {
                _stopwatch.Stop();
                
                var elapsedMs = _stopwatch.ElapsedMilliseconds;
                
                lock (_monitor._lock)
                {
                    if (!_monitor._methodMetrics.ContainsKey(_methodName))
                    {
                        _monitor._methodMetrics[_methodName] = new MethodMetrics
                        {
                            MethodName = _methodName,
                            TotalCalls = 1,
                            TotalTimeMs = elapsedMs,
                            MaxTimeMs = elapsedMs,
                            MinTimeMs = elapsedMs
                        };
                    }
                    else
                    {
                        var metrics = _monitor._methodMetrics[_methodName];
                        metrics.TotalCalls++;
                        metrics.TotalTimeMs += elapsedMs;
                        metrics.MaxTimeMs = Math.Max(metrics.MaxTimeMs, elapsedMs);
                        metrics.MinTimeMs = Math.Min(metrics.MinTimeMs, elapsedMs);
                    }
                }
                
                _monitor._logger.LogDebug("方法执行完成：{MethodName}，耗时：{ElapsedMs}ms，平均耗时：{AverageTimeMs:F2}ms",
                    _methodName, elapsedMs, _monitor.GetMethodStats(_methodName).AverageTimeMs);
            }
            
            _disposed = true;
        }
    }
}
```

### 7.2 DatabaseMonitor 类详细设计

```csharp
using System;
using System.Collections.Concurrent;
using System.Data.SqlClient;
using System.Diagnostics;
using System.Linq;
using Microsoft.Extensions.Logging;

namespace RUINORERP.Common.PerformanceMonitoring
{
    /// <summary>
    /// 数据库监控器 - 检测死锁和性能问题
    /// </summary>
    public class DatabaseMonitor : IDisposable
    {
        private readonly ILogger _logger;
        private readonly ConcurrentDictionary<string, QueryMetrics> _queryMetrics;
        private readonly ConcurrentDictionary<int, TransactionMetrics> _transactionMetrics;
        private readonly Timer _deadlockDetectionTimer;
        private readonly object _lock = new object();
        private bool _disposed = false;
        
        /// <summary>
        /// 查询指标
        /// </summary>
        public class QueryMetrics
        {
            public string QueryText { get; set; }
            public long ExecutionTimeMs { get; set; }
            public int ParameterCount { get; set; }
            public int RowCount { get; set; }
            public DateTime ExecuteTime { get; set; }
        }
        
        /// <summary>
        /// 事务指标
        /// </summary>
        public class TransactionMetrics
        {
            public int TransactionID { get; set; }
            public DateTime StartTime { get; set; }
            public DateTime? EndTime { get; set; }
            public long DurationMs => EndTime.HasValue ? (long)(EndTime.Value - StartTime).TotalMilliseconds : 0;
            public bool IsLongRunning => DurationMs > 30000; // 超过30秒
        }
        
        /// <summary>
        /// 数据库统计
        /// </summary>
        public class DatabaseStats
        {
            public List<QueryMetrics> SlowQueries { get; set; }
            public List<TransactionMetrics> LongRunningTransactions { get; set; }
            public int TotalQueries { get; set; }
            public int TotalTransactions { get; set; }
            public DateTime CollectTime { get; set; }
        }
        
        public DatabaseMonitor(ILogger<DatabaseMonitor> logger)
        {
            _logger = logger;
            _deadlockDetectionTimer = new Timer(DetectDeadlocks, null, TimeSpan.FromSeconds(30), TimeSpan.FromSeconds(30));
        }
        
        /// <summary>
        /// 监控 SQL 查询
        /// </summary>
        public IDisposable MonitorQuery(string query)
        {
            return new QueryPerformanceTracker(this, query);
        }
        
        /// <summary>
        /// 开始事务监控
        /// </summary>
        public int BeginTransaction()
        {
            var transactionId = DateTime.Now.Millisecond;
            lock (_lock)
            {
                _transactionMetrics[transactionId] = new TransactionMetrics
                {
                    TransactionID = transactionId,
                    StartTime = DateTime.Now
                };
            }
            return transactionId;
        }
        
        /// <summary>
        /// 结束事务监控
        /// </summary>
        public void EndTransaction(int transactionId)
        {
            lock (_lock)
            {
                if (_transactionMetrics.TryGetValue(transactionId, out var metrics))
                {
                    metrics.EndTime = DateTime.Now;
                }
            }
        }
        
        /// <summary>
        /// 获取数据库统计
        /// </summary>
        public DatabaseStats GetStats()
        {
            lock (_lock)
            {
                return new DatabaseStats
                {
                    SlowQueries = _queryMetrics.Values.Where(q => q.ExecutionTimeMs > 1000).ToList(),
                    LongRunningTransactions = _transactionMetrics.Values.Where(t => t.IsLongRunning).ToList(),
                    TotalQueries = _queryMetrics.Count,
                    TotalTransactions = _transactionMetrics.Count,
                    CollectTime = DateTime.Now
                };
            }
        }
        
        /// <summary>
        /// 检测死锁
        /// </summary>
        private void DetectDeadlocks(object state)
        {
            try
            {
                var stats = GetStats();
                
                if (stats.LongRunningTransactions.Count > 0)
                {
                    _logger.LogWarning("检测到 {Count} 个可能的事务死锁，事务ID：{TransactionIDs}",
                        stats.LongRunningTransactions.Count,
                        string.Join(", ", stats.LongRunningTransactions.Select(t => t.TransactionID)));
                }
                
                if (stats.SlowQueries.Count > 10)
                {
                    _logger.LogWarning("检测到 {Count} 个慢查询，平均耗时：{AverageTimeMs:F2}ms",
                        stats.SlowQueries.Count,
                        stats.SlowQueries.Average(q => q q.ExecutionTimeMs));
                }
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "检测死锁失败");
            }
        }
        
        /// <summary>
        /// 释放资源
        /// </summary>
        public void Dispose()
        {
            if (!_disposed)
            {
                _deadlockDetectionTimer?.Dispose();
                _disposed = true;
            }
        }
    }
    
    /// <summary>
    /// 查询性能跟踪器
    /// </summary>
    public class QueryPerformanceTracker : IDisposable
    {
        private readonly DatabaseMonitor _monitor;
        private readonly string _query;
        private readonly Stopwatch _stopwatch;
        private bool _disposed = false;
        
        public QueryPerformanceTracker(DatabaseMonitor monitor, string query)
        {
            _monitor = _monitor;
            _query = query;
            _stopwatch = Stopwatch.StartNew();
        }
        
        /// <summary>
        /// 释放资源
        /// </summary>
        public void Dispose()
        {
            if (!_disposed)
            {
                _stopwatch.Stop();
                
                var elapsedMs = _stopwatch.ElapsedMilliseconds;
                
                lock (_monitor._lock)
                {
                    _monitor._queryMetrics[_query] = new QueryMetrics
                    {
                        QueryText = _query,
                        ExecutionTimeMs = elapsedMs,
                        ExecuteTime = DateTime.Now
                    };
                }
                
                if (elapsedMs > 1000)
                {
                    _monitor._logger.LogWarning("慢查询执行完成，耗时：{ElapsedMs}ms，查询：{Query}",
                        elapsedMs, _query.Substring(0, Math.Min(100, _query.Length)));
                }
                else
                {
                    _monitor._logger.LogDebug("查询执行完成，耗时：{ElapsedMs}ms", elapsedMs);
                }
            }
            
            _disposed = true;
        }
    }
}
```
