public async Task<CascadeDeleteResult> ExecuteCascadeDeleteAsync<T>(
    List<long> targetIds, 
    bool isTestMode = false) where T : BaseEntity, new()
{
    var controller = Startup.GetFromFacByName<BaseController<T>>(typeof(T).Name + "Controller");
    
    // ✅ 使用事务包装,测试模式下会回滚
    using (var scope = new TransactionScope(TransactionScopeAsyncFlowOption.Enabled))
    {
        foreach (var id in targetIds)
        {
            var entity = Activator.CreateInstance<T>();
            entity.PrimaryKeyID = id;
            
            // ✅ 无论测试还是正式模式,都执行真正的删除操作
            bool success = await controller.BaseDeleteByNavAsync(entity);
            // BaseDeleteByNavAsync 内部调用:
            // _db.DeleteNav<T>().Include(m => m.RelatedEntities).ExecuteCommandAsync()
        }
        
        if (isTestMode)
        {
            // ✅ 测试模式:不调用 scope.Complete(),事务自动回滚
            Log("[测试模式] 事务已回滚,数据未实际删除");
        }
        else
        {
            // ✅ 正式模式:提交事务
            scope.Complete();
        }
    }
}
