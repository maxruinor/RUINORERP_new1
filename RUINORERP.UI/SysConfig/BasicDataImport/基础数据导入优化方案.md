# 基础数据导入功能优化方案

## 优化目标
基于 DevTools 中的优秀实践，优化基础数据导入功能，提升用户体验和数据导入效率。

---

## 一、DevTools 优秀思路分析

### 1.1 frmDbColumnToExlColumnConfig.cs 优秀特性

| 特性 | 说明 | 借鉴价值 |
|------|------|----------|
| **智能列匹配** | 使用编辑距离算法自动匹配 Excel 列和数据库字段 | 减少用户手动配置工作量 |
| **语义匹配** | 结合字段中文描述进行匹配 | 提高匹配准确率 |
| **主键智能识别** | 自动识别编号/代码类字段作为逻辑主键 | 简化主键配置 |
| **模板应用** | 支持导入模板快速应用 | 提高重复导入效率 |
| **可视化配置** | 双列表框拖拽匹配 | 直观易用 |

### 1.2 UCDataImportTool.cs 优秀特性

| 特性 | 说明 | 借鉴价值 |
|------|------|----------|
| **图片提取** | 从 Excel 中提取图片数据 | 支持产品图片导入 |
| **多格式支持** | 支持 Excel 和 CSV 格式 | 提高兼容性 |
| **简单直接** | 一键导入，操作简便 | 降低使用门槛 |

### 3.1 智能列匹配优化

#### 3.1.1 借鉴 DevTools 的智能匹配算法

**创建 SmartColumnMatcher 类：**

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using SqlSugar;

namespace RUINORERP.UI.SysConfig.BasicDataImport
{
    /// <summary>
    /// 智能列匹配器
    /// 借鉴 frmDbColumnToExlColumnConfig 的智能匹配算法
    /// </summary>
    public class SmartColumnMatcher
    {
        /// <summary>
        /// 执行智能列匹配
        /// </summary>
        /// <param name="excelColumns">Excel列名列表</param>
        /// <param name="entityType">目标实体类型</param>
        /// <param name="dbColumns">数据库列信息</param>
        /// <returns>匹配结果列表</returns>
        public List<ColumnMatchResult> MatchColumns(
            List<string> excelColumns, 
            Type entityType, 
            List<DbColumnInfo> dbColumns)
        {
            var results = new List<ColumnMatchResult>();
            var matchedExcelCols = new HashSet<string>();
            var matchedDbCols = new HashSet<string>();
            
            // 获取实体字段元数据
            var fieldInfos = GetEntityFieldInfos(entityType);
            
            foreach (var excelCol in excelColumns)
            {
                DbColumnInfo bestMatch = null;
                double maxScore = 0;
                
                foreach (var dbCol in dbColumns)
                {
                    if (matchedDbCols.Contains(dbCol.ColumnName))
                        continue;
                    
                    double score = CalculateMatchScore(excelCol, dbCol, fieldInfos);
                    if (score > maxScore && score > 0.65) // 阈值0.65
                    {
                        maxScore = score;
                        bestMatch = dbCol;
                    }
                }
                
                if (bestMatch != null)
                {
                    results.Add(new ColumnMatchResult
                    {
                        ExcelColumn = excelCol,
                        DbColumn = bestMatch.ColumnName,
                        Score = maxScore,
                        IsPrimaryKey = IsPrimaryKeyCandidate(bestMatch, fieldInfos),
                        DataType = bestMatch.DataType
                    });
                    
                    matchedExcelCols.Add(excelCol);
                    matchedDbCols.Add(bestMatch.ColumnName);
                }
            }
            
            return results.OrderByDescending(r => r.Score).ToList();
        }
        
        /// <summary>
        /// 计算匹配分数（综合算法）
        /// </summary>
        private double CalculateMatchScore(string excelCol, DbColumnInfo dbCol, List<FieldInfo> fieldInfos)
        {
            // 1. 精确匹配（最高分）
            if (excelCol.Equals(dbCol.ColumnName, StringComparison.OrdinalIgnoreCase))
                return 1.0;
            if (excelCol.Equals(dbCol.Description, StringComparison.OrdinalIgnoreCase))
                return 1.0;
            
            // 2. 包含匹配
            if (dbCol.Description?.Contains(excelCol) == true || 
                excelCol.Contains(dbCol.Description ?? ""))
                return 0.9;
            
            // 3. 编辑距离相似度（列名）
            double nameScore = CalculateSimilarity(excelCol.ToLower(), dbCol.ColumnName.ToLower());
            
            // 4. 编辑距离相似度（描述）
            double descScore = 0;
            if (!string.IsNullOrEmpty(dbCol.Description))
            {
                descScore = CalculateSimilarity(excelCol.ToLower(), dbCol.Description.ToLower());
            }
            
            // 5. 字段元数据增强
            var fieldInfo = fieldInfos?.FirstOrDefault(f => 
                f.ColumnName == dbCol.ColumnName || f.PropertyName == dbCol.ColumnName);
            
            double metadataBoost = 0;
            if (fieldInfo != null)
            {
                // 如果字段描述与 Excel 列名相似，增加权重
                if (!string.IsNullOrEmpty(fieldInfo.Description))
                {
                    metadataBoost = CalculateSimilarity(excelCol.ToLower(), fieldInfo.Description.ToLower()) * 0.1;
                }
            }
            
            return Math.Max(nameScore, descScore) * 0.8 + metadataBoost;
        }
        
        /// <summary>
        /// 计算字符串相似度（基于编辑距离）
        /// </summary>
        private double CalculateSimilarity(string s, string t)
        {
            if (string.IsNullOrEmpty(s) || string.IsNullOrEmpty(t))
                return 0;
            
            int distance = LevenshteinDistance(s, t);
            int maxLen = Math.Max(s.Length, t.Length);
            
            return maxLen == 0 ? 1.0 : 1.0 - (double)distance / maxLen;
        }
        
        /// <summary>
        /// 莱文斯坦距离（编辑距离）
        /// </summary>
        private int LevenshteinDistance(string s, string t)
        {
            int n = s.Length;
            int m = t.Length;
            int[,] d = new int[n + 1, m + 1];
            
            if (n == 0) return m;
            if (m == 0) return n;
            
            for (int i = 0; i <= n; d[i, 0] = i++) { }
            for (int j = 0; j <= m; d[0, j] = j++) { }
            
            for (int i = 1; i <= n; i++)
            {
                for (int j = 1; j <= m; j++)
                {
                    int cost = (t[j - 1] == s[i - 1]) ? 0 : 1;
                    d[i, j] = Math.Min(
                        Math.Min(d[i - 1, j] + 1, d[i, j - 1] + 1),
                        d[i - 1, j - 1] + cost);
                }
            }
            
            return d[n, m];
        }
        
        /// <summary>
        /// 判断是否为主键候选字段
        /// </summary>
        private bool IsPrimaryKeyCandidate(DbColumnInfo dbCol, List<FieldInfo> fieldInfos)
        {
            string[] keyWords = { "code", "no", "number", "编号", "代码", "单号", "id", "key" };
            string checkStr = (dbCol.ColumnName + " " + dbCol.Description).ToLower();
            
            // 检查关键字
            bool hasKeyword = keyWords.Any(k => checkStr.Contains(k));
            
            // 检查实体元数据
            var fieldInfo = fieldInfos?.FirstOrDefault(f => 
                f.ColumnName == dbCol.ColumnName || f.PropertyName == dbCol.ColumnName);
            
            bool isPrimaryKey = fieldInfo?.IsPrimaryKey ?? false;
            
            return hasKeyword || isPrimaryKey;
        }
        
        /// <summary>
        /// 获取实体字段元数据
        /// </summary>
        private List<FieldInfo> GetEntityFieldInfos(Type entityType)
        {
            if (entityType == null) return null;
            
            var infos = new List<FieldInfo>();
            var properties = entityType.GetProperties();
            
            foreach (var prop in properties)
            {
                var sugarCol = prop.GetCustomAttribute<SugarColumn>();
                if (sugarCol != null)
                {
                    infos.Add(new FieldInfo
                    {
                        PropertyName = prop.Name,
                        ColumnName = sugarCol.ColumnName ?? prop.Name,
                        Description = sugarCol.ColumnDescription,
                        IsPrimaryKey = sugarCol.IsPrimaryKey,
                        DataType = sugarCol.ColumnDataType
                    });
                }
            }
            
            return infos;
        }
    }
    
    /// <summary>
    /// 列匹配结果
    /// </summary>
    public class ColumnMatchResult
    {
        public string ExcelColumn { get; set; }
        public string DbColumn { get; set; }
        public double Score { get; set; }
        public bool IsPrimaryKey { get; set; }
        public string DataType { get; set; }
    }
    
    /// <summary>
    /// 数据库列信息
    /// </summary>
    public class DbColumnInfo
    {
        public string ColumnName { get; set; }
        public string DataType { get; set; }
        public string Description { get; set; }
        public bool IsNullable { get; set; }
    }
    
    /// <summary>
    /// 字段元数据
    /// </summary>
    public class FieldInfo
    {
        public string PropertyName { get; set; }
        public string ColumnName { get; set; }
        public string Description { get; set; }
        public bool IsPrimaryKey { get; set; }
        public string DataType { get; set; }
    }
}
```

### 3.2 导入模板管理优化

#### 3.2.1 创建 ImportTemplateManager 类

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Xml.Serialization;

namespace RUINORERP.UI.SysConfig.BasicDataImport
{
    /// <summary>
    /// 导入模板管理器
    /// 借鉴 DevTools 的模板应用思路
    /// </summary>
    public class ImportTemplateManager
    {
        private readonly string _templateDirectory;
        private readonly XmlSerializer _serializer;
        
        public ImportTemplateManager()
        {
            _templateDirectory = Path.Combine(
                AppDomain.CurrentDomain.BaseDirectory, 
                "ImportTemplates");
            
            if (!Directory.Exists(_templateDirectory))
            {
                Directory.CreateDirectory(_templateDirectory);
            }
            
            _serializer = new XmlSerializer(typeof(ImportTemplate));
        }
        
        /// <summary>
        /// 保存导入模板
        /// </summary>
        public void SaveTemplate(ImportTemplate template)
        {
            if (template == null) throw new ArgumentNullException(nameof(template));
            
            string filePath = Path.Combine(_templateDirectory, $"{template.TemplateName}.xml");
            
            using (var writer = new StreamWriter(filePath))
            {
                _serializer.Serialize(writer, template);
            }
        }
        
        /// <summary>
        /// 加载导入模板
        /// </summary>
        public ImportTemplate LoadTemplate(string templateName)
        {
            string filePath = Path.Combine(_templateDirectory, $"{templateName}.xml");
            
            if (!File.Exists(filePath))
                return null;
            
            using (var reader = new StreamReader(filePath))
            {
                return (ImportTemplate)_serializer.Deserialize(reader);
            }
        }
        
        /// <summary>
        /// 获取所有模板名称
        /// </summary>
        public List<string> GetAllTemplateNames()
        {
            return Directory.GetFiles(_templateDirectory, "*.xml")
                .Select(Path.GetFileNameWithoutExtension)
                .ToList();
        }
        
        /// <summary>
        /// 根据实体类型获取适用模板
        /// </summary>
        public List<ImportTemplate> GetTemplatesForEntity(Type entityType)
        {
            var templates = new List<ImportTemplate>();
            
            foreach (var templateName in GetAllTemplateNames())
            {
                var template = LoadTemplate(templateName);
                if (template?.EntityType == entityType.FullName)
                {
                    templates.Add(template);
                }
            }
            
            return templates;
        }
        
        /// <summary>
        /// 应用模板到配置
        /// </summary>
        public void ApplyTemplate(ImportConfiguration config, ImportTemplate template)
        {
            if (template?.ColumnMappings == null) return;
            
            foreach (var mapping in template.ColumnMappings)
            {
                var existingMapping = config.ColumnMappings?.FirstOrDefault(
                    m => m.SystemField?.Key == mapping.DbColumn);
                
                if (existingMapping != null)
                {
                    existingMapping.ExcelColumn = mapping.ExcelColumn;
                }
            }
            
            // 标记逻辑主键
            if (!string.IsNullOrEmpty(template.LogicalKeyField))
            {
                var keyMapping = config.ColumnMappings?.FirstOrDefault(
                    m => m.SystemField?.Key == template.LogicalKeyField);
                if (keyMapping != null)
                {
                    keyMapping.IsUniqueValue = true;
                }
            }
        }
    }
    
    /// <summary>
    /// 导入模板
    /// </summary>
    [Serializable]
    public class ImportTemplate
    {
        public string TemplateName { get; set; }
        public string EntityType { get; set; }
        public string Description { get; set; }
        public List<TemplateColumnMapping> ColumnMappings { get; set; }
        public string LogicalKeyField { get; set; }
        public DateTime CreateTime { get; set; }
        public DateTime UpdateTime { get; set; }
    }
    
    /// <summary>
    /// 模板列映射
    /// </summary>
    [Serializable]
    public class TemplateColumnMapping
    {
        public string ExcelColumn { get; set; }
        public string DbColumn { get; set; }
        public bool IsKey { get; set; }
    }
}
```

### 3.3 图片提取功能集成

#### 3.3.1 借鉴 UCDataImportTool 的图片提取思路

**优化 DynamicExcelParser 类：**

```csharp
using System;
using System.Collections.Generic;
using System.Data;
using System.IO;
using System.Linq;
using NPOI.SS.UserModel;

namespace RUINORERP.UI.SysConfig.BasicDataImport
{
    /// <summary>
    /// 增强型 Excel 解析器
    /// 借鉴 UCDataImportTool 的图片提取功能
    /// </summary>
    public class EnhancedExcelParser
    {
        /// <summary>
        /// 解析 Excel 文件（支持图片提取）
        /// </summary>
        public ExcelParseResult ParseWithImages(string filePath, int sheetIndex = 0)
        {
            var result = new ExcelParseResult
            {
                DataTable = ParseExcel(filePath, sheetIndex),
                Images = new Dictionary<int, List<ExcelImageInfo>>()
            };
            
            // 提取图片
            try
            {
                var imageMap = ExtractImagesFromExcel(filePath, sheetIndex);
                result.Images = imageMap;
                
                // 将图片关联到数据行
                AssociateImagesWithData(result);
            }
            catch (Exception ex)
            {
                // 图片提取失败不影响主流程
                System.Diagnostics.Debug.WriteLine($"图片提取失败: {ex.Message}");
            }
            
            return result;
        }
        
        /// <summary>
        /// 从 Excel 提取图片
        /// </summary>
        private Dictionary<int, List<ExcelImageInfo>> ExtractImagesFromExcel(string filePath, int sheetIndex)
        {
            var imageMap = new Dictionary<int, List<ExcelImageInfo>>();
            
            using (var fs = new FileStream(filePath, FileMode.Open, FileAccess.Read))
            {
                var workbook = WorkbookFactory.Create(fs);
                var sheet = workbook.GetSheetAt(sheetIndex);
                
                // 获取所有图片
                var pictures = sheet.Workbook.GetAllPictures();
                
                // 遍历绘图对象
                var drawing = sheet.DrawingPatriarch;
                if (drawing != null)
                {
                    foreach (var shape in drawing.GetShapes())
                    {
                        if (shape is IPicture picture)
                        {
                            var anchor = picture.ClientAnchor;
                            int row1 = anchor.Row1;
                            
                            var imageInfo = new ExcelImageInfo
                            {
                                RowIndex = row1,
                                ColumnIndex = anchor.Col1,
                                ImageData = picture.PictureData.Data,
                                ImageType = picture.PictureData.SuggestFileExtension(),
                                Width = picture.ImageDimension.Width,
                                Height = picture.ImageDimension.Height
                            };
                            
                            if (!imageMap.ContainsKey(row1))
                            {
                                imageMap[row1] = new List<ExcelImageInfo>();
                            }
                            imageMap[row1].Add(imageInfo);
                        }
                    }
                }
            }
            
            return imageMap;
        }
        
        /// <summary>
        /// 将图片关联到数据行
        /// </summary>
        private void AssociateImagesWithData(ExcelParseResult result)
        {
            if (!result.DataTable.Columns.Contains("_Images"))
            {
                result.DataTable.Columns.Add("_Images", typeof(List<ExcelImageInfo>));
            }
            
            // 假设标题行占1行，数据从第2行开始（索引1）
            int headerOffset = 1;
            
            foreach (DataRow row in result.DataTable.Rows)
            {
                int rowIndex = result.DataTable.Rows.IndexOf(row) + headerOffset;
                
                if (result.Images.ContainsKey(rowIndex))
                {
                    row["_Images"] = result.Images[rowIndex];
                }
            }
        }
        
        /// <summary>
        /// 基础 Excel 解析
        /// </summary>
        private DataTable ParseExcel(string filePath, int sheetIndex)
        {
            // 复用现有的 DynamicExcelParser 逻辑
            var parser = new DynamicExcelParser();
            return parser.Parse(filePath, sheetIndex);
        }
    }
    
    /// <summary>
    /// Excel 解析结果
    /// </summary>
    public class ExcelParseResult
    {
        public DataTable DataTable { get; set; }
        public Dictionary<int, List<ExcelImageInfo>> Images { get; set; }
    }
    
    /// <summary>
    /// Excel 图片信息
    /// </summary>
    public class ExcelImageInfo
    {
        public int RowIndex { get; set; }
        public int ColumnIndex { get; set; }
        public byte[] ImageData { get; set; }
        public string ImageType { get; set; }
        public double Width { get; set; }
        public double Height { get; set; }
    }
}
```

### 3.4 列映射配置界面优化

#### 3.4.1 借鉴 DevTools 的可视化配置

**优化 frmColumnMappingConfig 类：**

```csharp
using System;
using System.Collections.Generic;
using System.Drawing;
using System.Linq;
using System.Windows.Forms;
using Krypton.Toolkit;

namespace RUINORERP.UI.SysConfig.BasicDataImport
{
    /// <summary>
    /// 优化的列映射配置窗体
    /// 借鉴 frmDbColumnToExlColumnConfig 的可视化设计
    /// </summary>
    public partial class frmColumnMappingConfig : KryptonForm
    {
        private SmartColumnMatcher _matcher;
        private ImportTemplateManager _templateManager;
        private List<ColumnMatchResult> _matchResults;
        
        public frmColumnMappingConfig()
        {
            InitializeComponent();
            _matcher = new SmartColumnMatcher();
            _templateManager = new ImportTemplateManager();
            _matchResults = new List<ColumnMatchResult>();
        }
        
        /// <summary>
        /// 执行智能匹配
        /// </summary>
        private void ExecuteSmartMatch()
        {
            var excelColumns = GetExcelColumns();
            var dbColumns = GetDbColumns();
            var entityType = GetSelectedEntityType();
            
            _matchResults = _matcher.MatchColumns(excelColumns, entityType, dbColumns);
            
            // 显示匹配结果
            DisplayMatchResults(_matchResults);
            
            // 高亮主键
            HighlightPrimaryKeys();
            
            // 显示匹配统计
            ShowMatchStatistics(_matchResults.Count, excelColumns.Count);
        }
        
        /// <summary>
        /// 显示匹配结果
        /// </summary>
        private void DisplayMatchResults(List<ColumnMatchResult> results)
        {
            kryptonDataGridViewMatches.Rows.Clear();
            
            foreach (var result in results)
            {
                int rowIndex = kryptonDataGridViewMatches.Rows.Add(
                    result.ExcelColumn,
                    result.DbColumn,
                    $"{(result.Score * 100):F1}%",
                    result.IsPrimaryKey ? "是" : "否",
                    result.DataType
                );
                
                // 高亮高置信度匹配
                if (result.Score > 0.9)
                {
                    kryptonDataGridViewMatches.Rows[rowIndex].DefaultCellStyle.BackColor = Color.LightGreen;
                }
                else if (result.Score > 0.75)
                {
                    kryptonDataGridViewMatches.Rows[rowIndex].DefaultCellStyle.BackColor = Color.LightYellow;
                }
            }
        }
        
        /// <summary>
        /// 应用导入模板
        /// </summary>
        private void ApplyImportTemplate(string templateName)
        {
            var template = _templateManager.LoadTemplate(templateName);
            if (template == null)
            {
                KryptonMessageBox.Show("模板不存在", "提示", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }
            
            // 应用模板到当前配置
            _templateManager.ApplyTemplate(_currentConfig, template);
            
            // 刷新显示
            RefreshMappingDisplay();
            
            MainForm.Instance.PrintInfoLog($"已应用模板 [{template.TemplateName}]");
        }
        
        /// <summary>
        /// 保存为模板
        /// </summary>
        private void SaveAsTemplate()
        {
            var templateName = KryptonInputBox.Show("请输入模板名称:", "保存模板", "新模板");
            if (string.IsNullOrEmpty(templateName)) return;
            
            var template = new ImportTemplate
            {
                TemplateName = templateName,
                EntityType = _selectedEntityType?.FullName,
                Description = $"{_selectedEntityType?.Name} 导入模板",
                ColumnMappings = _currentConfig.ColumnMappings?.Select(m => new TemplateColumnMapping
                {
                    ExcelColumn = m.ExcelColumn,
                    DbColumn = m.SystemField?.Key,
                    IsKey = m.IsUniqueValue
                }).ToList(),
                LogicalKeyField = _currentConfig.ColumnMappings?.FirstOrDefault(m => m.IsUniqueValue)?.SystemField?.Key,
                CreateTime = DateTime.Now,
                UpdateTime = DateTime.Now
            };
            
            _templateManager.SaveTemplate(template);
            
            // 刷新模板列表
            LoadTemplateList();
            
            KryptonMessageBox.Show("模板保存成功", "提示", MessageBoxButtons.OK, MessageBoxIcon.Information);
        }
        
        /// <summary>
        /// 加载模板列表
        /// </summary>
        private void LoadTemplateList()
        {
            kcmbTemplates.Items.Clear();
            
            var templates = _templateManager.GetAllTemplateNames();
            foreach (var templateName in templates)
            {
                kcmbTemplates.Items.Add(templateName);
            }
        }
    }
}
```
