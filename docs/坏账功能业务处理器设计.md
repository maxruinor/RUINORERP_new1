## 1. 业务处理器架构

### 1.1 处理器设计模式

坏账业务处理器遵循现有的业务处理器设计模式：
- 继承 `IEntityBaseRepository<tb_FM_BadDebt>` 接口
- 使用依赖注入获取相关仓储
- 支持事务操作
- 统一的返回值格式 `OperationResult<T>`

## 2. 核心业务方法详解

### 2.1 创建坏账申请 (CreateAsync)

```csharp
/// <summary>
/// 创建坏账申请
/// </summary>
/// <param name="badDebt">坏账申请实体</param>
/// <returns>创建结果</returns>
public async Task<OperationResult<tb_FM_BadDebt>> CreateAsync(tb_FM_BadDebt badDebt)
{
    try
    {
        // 1. 基本校验
        if (badDebt == null)
        {
            return OperationResult<tb_FM_BadDebt>.Fail("坏账申请数据不能为空");
        }

        // 2. 如果关联了应收/应付单，进行校验
        if (badDebt.ARAPId.HasValue)
        {
            var arap = await _arapRepo.FindAsync(badDebt.ARAPId.Value);
            if (arap == null)
            {
                return OperationResult<tb_FM_BadDebt>.Fail("关联的应收/应付单不存在");
            }

            // 3. 获取可申请坏账金额
            var availableAmount = await GetAvailableBadDebtAmountAsync(badDebt.ARAPId.Value);
            if (badDebt.BadDebtAmount > availableAmount)
            {
                return OperationResult<tb_FM_BadDebt>.Fail(
                    $"坏账金额({badDebt.BadDebtAmount})不能超过可申请金额({availableAmount})");
            }

            // 4. 填充关联信息
            badDebt.ARAPNo = arap.ARAPNo;
            badDebt.SourceBillId = arap.SourceBillId;
            badDebt.SourceBillNo = arap.SourceBillNo;
            badDebt.SourceBizType = arap.SourceBizType;
            badDebt.CustomerVendor_ID = arap.CustomerVendor_ID;
            badDebt.Currency_ID = arap.Currency_ID;
            badDebt.ExchangeRate = arap.ExchangeRate;
            badDebt.OriginalAmount = arap.TotalLocalPayableAmount;
            badDebt.UnpaidAmount = arap.LocalBalanceAmount;
            badDebt.ReceivePaymentType = arap.ReceivePaymentType;
        }

        // 5. 计算剩余坏账金额
        badDebt.RemainingAmount = badDebt.BadDebtAmount;
        badDebt.ReversedAmount = 0;

        // 6. 设置初始状态
        badDebt.BadDebtStatus = (int)BadDebtStatus.草稿;
        badDebt.DataStatus = (int)DataStatus.草稿;
        badDebt.Created_at = DateTime.Now;
        badDebt.Created_by = _context.GetCurrentUserId();

        // 7. 生成单号
        badDebt.BadDebtNo = await _sequenceService.GenerateAsync("BadDebt", "BD");

        // 8. 保存
        var result = await _badDebtRepo.InsertAsync(badDebt);

        return OperationResult<tb_FM_BadDebt>.Success(result);
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, "创建坏账申请失败");
        return OperationResult<tb_FM_BadDebt>.Fail($"创建坏账申请失败: {ex.Message}");
    }
}
```

### 2.2 提交坏账申请 (SubmitAsync)

```csharp
/// <summary>
/// 提交坏账申请
/// </summary>
/// <param name="badDebtId">坏账申请Id</param>
/// <returns>提交结果</returns>
public async Task<OperationResult<tb_FM_BadDebt>> SubmitAsync(long badDebtId)
{
    try
    {
        var badDebt = await _badDebtRepo.FindAsync(badDebtId);
        if (badDebt == null)
        {
            return OperationResult<tb_FM_BadDebt>.Fail("坏账申请不存在");
        }

        // 校验状态
        if (badDebt.BadDebtStatus != (int)BadDebtStatus.草稿)
        {
            return OperationResult<tb_FM_BadDebt>.Fail("只有草稿状态的坏账申请才能提交");
        }

        // 数据校验
        var validationResult = await ValidateAsync(badDebt);
        if (!validationResult.Success)
        {
            return validationResult;
        }

        // 更新状态
        badDebt.BadDebtStatus = (int)BadDebtStatus.待审核;
        badDebt.DataStatus = (int)DataStatus.新建;
        badDebt.Modified_at = DateTime.Now;
        badDebt.Modified_by = _context.GetCurrentUserId();

        await _badDebtRepo.UpdateAsync(badDebt);

        return OperationResult<tb_FM_BadDebt>.Success(badDebt);
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"提交坏账申请失败, Id={badDebtId}");
        return OperationResult<tb_FM_BadDebt>.Fail($"提交失败: {ex.Message}");
    }
}
```

### 2.3 审核坏账申请 (ApproveAsync) - 核销处理

```csharp
/// <summary>
/// 审核坏账申请
/// 审核通过后执行核销：扣减应收/应付单余额，生成核销记录
/// </summary>
/// <param name="badDebtId">坏账申请Id</param>
/// <param name="approvalOpinions">审批意见</param>
/// <returns>审核结果</returns>
public async Task<OperationResult<tb_FM_BadDebt>> ApproveAsync(long badDebtId, string approvalOpinions)
{
    using (var db = _badDebtRepo.GetDatabase())
    {
        db.BeginTran();
        try
        {
            var badDebt = await _badDebtRepo.FindAsync(badDebtId);
            if (badDebt == null)
            {
                db.RollbackTran();
                return OperationResult<tb_FM_BadDebt>.Fail("坏账申请不存在");
            }

            // 校验状态
            if (badDebt.BadDebtStatus != (int)BadDebtStatus.待审核)
            {
                db.RollbackTran();
                return OperationResult<tb_FM_BadDebt>.Fail("只有待审核状态的坏账申请才能审核");
            }

            // 核销应收/应付单
            if (badDebt.ARAPId.HasValue)
            {
                var arap = await _arapRepo.FindAsync(badDebt.ARAPId.Value);
                if (arap == null)
                {
                    db.RollbackTran();
                    return OperationResult<tb_FM_BadDebt>.Fail("关联的应收/应付单不存在");
                }

                // 再次校验余额
                var availableAmount = await GetAvailableBadDebtAmountAsync(badDebt.ARAPId.Value);
                if (badDebt.BadDebtAmount > availableAmount + badDebt.BadDebtAmount)
                {
                    db.RollbackTran();
                    return OperationResult<tb_FM_BadDebt>.Fail("坏账金额已超过可用金额，请重新申请");
                }

                // 记录原始余额
                decimal originalBalance = arap.LocalBalanceAmount;
                decimal afterBalance = originalBalance - badDebt.BadDebtAmount;

                // 更新应收/应付单
                arap.LocalBalanceAmount = afterBalance;
                arap.LocalPaidAmount = arap.LocalPaidAmount + badDebt.BadDebtAmount;
                arap.ARAPStatus = (int)ARAPStatus.坏账;  // 更新为坏账状态

                // 更新汇率相关信息
                if (badDebt.BadDebtForeignAmount.HasValue)
                {
                    arap.ForeignBalanceAmount = arap.ForeignBalanceAmount - badDebt.BadDebtForeignAmount.Value;
                }

                await _arapRepo.UpdateAsync(arap);

                // 创建核销记录
                var writeOff = new tb_FM_BadDebtWriteOff
                {
                    WriteOffId = _sequenceService.NextId(),
                    WriteOffNo = await _sequenceService.GenerateAsync("BadDebtWriteOff", "WO"),
                    BadDebtId = badDebt.BadDebtId,
                    BadDebtNo = badDebt.BadDebtNo,
                    ARAPId = arap.ARAPId,
                    ARAPNo = arap.ARAPNo,
                    ReceivePaymentType = badDebt.ReceivePaymentType,
                    CustomerVendor_ID = badDebt.CustomerVendor_ID,
                    Currency_ID = badDebt.Currency_ID,
                    ExchangeRate = badDebt.ExchangeRate,
                    OriginalLocalBalance = originalBalance,
                    WriteOffLocalAmount = badDebt.BadDebtAmount,
                    WriteOffForeignAmount = badDebt.BadDebtForeignAmount,
                    AfterWriteOffLocalBalance = afterBalance,
                    SettlementType = (int)SettlementType.坏账核销,
                    IsReversedWriteOff = false,
                    WriteOffDate = DateTime.Now,
                    Employee_ID = badDebt.Employee_ID,
                    Remark = $"坏账核销：{badDebt.BadDebtNo}，原因：{badDebt.BadDebtReason}",
                    Created_at = DateTime.Now,
                    Created_by = _context.GetCurrentUserId()
                };

                await _writeOffRepo.InsertAsync(writeOff);
            }

            // 更新坏账申请状态
            badDebt.BadDebtStatus = (int)BadDebtStatus.已审核;
            badDebt.DataStatus = (int)DataStatus.确认;
            badDebt.ApprovalStatus = (int)ApprovalStatus.已审核;
            badDebt.ApprovalResults = true;
            badDebt.ApprovalOpinions = approvalOpinions;
            badDebt.Approver_by = _context.GetCurrentUserId();
            badDebt.Approver_at = DateTime.Now;
            badDebt.ProcessedAt = DateTime.Now;
            badDebt.IsWriteOff = true;
            badDebt.WriteOffAt = DateTime.Now;
            badDebt.Modified_at = DateTime.Now;
            badDebt.Modified_by = _context.GetCurrentUserId();

            await _badDebtRepo.UpdateAsync(badDebt);

            // 记录审计日志
            await _auditLogService.LogAsync("坏账审核", badDebt, "审核通过");

            db.CommitTran();
            return OperationResult<tb_FM_BadDebt>.Success(badDebt);
        }
        catch (Exception ex)
        {
            db.RollbackTran();
            _logger.LogError(ex, $"审核坏账申请失败, Id={badDebtId}");
            return OperationResult<tb_FM_BadDebt>.Fail($"审核失败: {ex.Message}");
        }
    }
}
```

### 2.4 撤销坏账 (ReverseBadDebtAsync) - 核心撤销逻辑

```csharp
/// <summary>
/// 撤销坏账
/// 业务场景：客户之前无法付款做了坏账处理，后来客户又愿意支付了
/// 撤销后：恢复应收/应付单余额，生成红字核销记录，状态变为"已撤销"
/// </summary>
/// <param name="badDebtId">坏账申请Id</param>
/// <param name="reverseReason">撤销原因</param>
/// <returns>撤销结果</returns>
public async Task<OperationResult<tb_FM_BadDebt>> ReverseBadDebtAsync(long badDebtId, string reverseReason)
{
    if (string.IsNullOrWhiteSpace(reverseReason))
    {
        return OperationResult<tb_FM_BadDebt>.Fail("撤销原因不能为空");
    }

    using (var db = _badDebtRepo.GetDatabase())
    {
        db.BeginTran();
        try
        {
            var badDebt = await _badDebtRepo.FindAsync(badDebtId);
            if (badDebt == null)
            {
                db.RollbackTran();
                return OperationResult<tb_FM_BadDebt>.Fail("坏账申请不存在");
            }

            // 校验状态：必须是已审核状态才能撤销
            if (badDebt.BadDebtStatus != (int)BadDebtStatus.已审核)
            {
                db.RollbackTran();
                return OperationResult<tb_FM_BadDebt>.Fail("只有已审核状态的坏账申请才能撤销");
            }

            // 校验是否已撤销
            if (badDebt.IsReversed)
            {
                db.RollbackTran();
                return OperationResult<tb_FM_BadDebt>.Fail("该坏账申请已经撤销，不能重复撤销");
            }

            // 恢复应收/应付单
            if (badDebt.ARAPId.HasValue)
            {
                var arap = await _arapRepo.FindAsync(badDebt.ARAPId.Value);
                if (arap == null)
                {
                    db.RollbackTran();
                    return OperationResult<tb_FM_BadDebt>.Fail("关联的应收/应付单不存在");
                }

                // 校验应收/应付单状态必须是坏账
                if (arap.ARAPStatus != (int)ARAPStatus.坏账)
                {
                    db.RollbackTran();
                    return OperationResult<tb_FM_BadDebt>.Fail("关联的应收/应付单状态不是坏账，无法撤销");
                }

                // 记录原始余额
                decimal originalBalance = arap.LocalBalanceAmount;
                // 恢复余额：加回之前核销的金额
                decimal afterBalance = originalBalance + badDebt.BadDebtAmount;

                // 恢复应收/应付单
                arap.LocalBalanceAmount = afterBalance;
                // 减少已核销金额
                arap.LocalPaidAmount = arap.LocalPaidAmount - badDebt.BadDebtAmount;
                // 恢复原状态为"待支付"（可以根据业务需要调整为其他状态）
                arap.ARAPStatus = (int)ARAPStatus.待支付;

                // 恢复外币余额
                if (badDebt.BadDebtForeignAmount.HasValue)
                {
                    arap.ForeignBalanceAmount = arap.ForeignBalanceAmount + badDebt.BadDebtForeignAmount.Value;
                }

                await _arapRepo.UpdateAsync(arap);

                // 创建红字核销记录（撤销核销）
                var reverseWriteOff = new tb_FM_BadDebtWriteOff
                {
                    WriteOffId = _sequenceService.NextId(),
                    WriteOffNo = await _sequenceService.GenerateAsync("BadDebtWriteOff", "WO"),
                    BadDebtId = badDebt.BadDebtId,
                    BadDebtNo = badDebt.BadDebtNo,
                    ARAPId = arap.ARAPId,
                    ARAPNo = arap.ARAPNo,
                    ReceivePaymentType = badDebt.ReceivePaymentType,
                    CustomerVendor_ID = badDebt.CustomerVendor_ID,
                    Currency_ID = badDebt.Currency_ID,
                    ExchangeRate = badDebt.ExchangeRate,
                    OriginalLocalBalance = originalBalance,
                    // 关键：负数表示撤销
                    WriteOffLocalAmount = -badDebt.BadDebtAmount,
                    WriteOffForeignAmount = badDebt.BadDebtForeignAmount.HasValue
                        ? -badDebt.BadDebtForeignAmount.Value : null,
                    AfterWriteOffLocalBalance = afterBalance,
                    SettlementType = (int)SettlementType.坏账撤销核销,
                    IsReversedWriteOff = true,
                    WriteOffDate = DateTime.Now,
                    Employee_ID = _context.GetCurrentUserId(),
                    Remark = $"撤销坏账核销：{badDebt.BadDebtNo}，撤销原因：{reverseReason}",
                    Created_at = DateTime.Now,
                    Created_by = _context.GetCurrentUserId()
                };

                await _writeOffRepo.InsertAsync(reverseWriteOff);
            }

            // 更新坏账申请状态
            badDebt.BadDebtStatus = (int)BadDebtStatus.已撤销;
            badDebt.DataStatus = (int)DataStatus.新建;
            badDebt.IsReversed = true;
            badDebt.ReversedAt = DateTime.Now;
            badDebt.ReversedBy = _context.GetCurrentUserId();
            badDebt.ReversedReason = reverseReason;
            badDebt.Modified_at = DateTime.Now;
            badDebt.Modified_by = _context.GetCurrentUserId();

            await _badDebtRepo.UpdateAsync(badDebt);

            // 记录审计日志
            await _auditLogService.LogAsync("撤销坏账", badDebt,
                $"撤销原因：{reverseReason}，恢复金额：{badDebt.BadDebtAmount}");

            db.CommitTran();

            _logger.LogInformation($"撤销坏账成功，BadDebtId={badDebtId}，撤销原因={reverseReason}");

            return OperationResult<tb_FM_BadDebt>.Success(badDebt);
        }
        catch (Exception ex)
        {
            db.RollbackTran();
            _logger.LogError(ex, $"撤销坏账失败, Id={badDebtId}");
            return OperationResult<tb_FM_BadDebt>.Fail($"撤销失败: {ex.Message}");
        }
    }
}
```

### 2.5 反审核 (ReverseApproveAsync)

```csharp
/// <summary>
/// 反审核坏账申请
/// 将已审核的坏账申请打回待审核状态，不恢复应收/应付单余额
/// </summary>

{
    try
    {
        var badDebt = await _badDebtRepo.FindAsync(badDebtId);
        if (badDebt == null)
        {
            return OperationResult<tb_FM_BadDebt>.Fail("坏账申请不存在");

        return OperationResult<tb_FM_BadDebt>.Success(badDebt);
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"反审核坏账申请失败, Id={badDebtId}");
```

## 3. 数据校验方法

### 3.1 ValidateAsync - 数据校验

```csharp
/// <summary>
/// 校验坏账申请数据
/// </summary>
/// <param name="badDebt">坏账申请实体</param>
/// <returns>校验结果</returns>
public async Task<OperationResult> ValidateAsync(tb_FM_BadDebt badDebt)
{
    // 坏账金额必须大于0
    if (badDebt.BadDebtAmount <= 0)
    {
        return OperationResult.Fail("坏账金额必须大于0");
    }

    // 必须选择坏账原因类型
    if (badDebt.BadDebtReasonType <= 0)
    {
        return OperationResult.Fail("请选择坏账原因类型");
    }

    // 必须选择经办人
    if (badDebt.Employee_ID <= 0)
    {
        return OperationResult.Fail("请选择经办人");
    }

    // 必须选择往来单位
    if (badDebt.CustomerVendor_ID <= 0)
    {
        return OperationResult.Fail("请选择往来单位");
    }

    // 必须选择币别
    if (badDebt.Currency_ID <= 0)
    {
        return OperationResult.Fail("请选择币别");
    }

    // 如果关联了应收/应付单，校验金额
    if (badDebt.ARAPId.HasValue)
    {
        var arap = await _arapRepo.FindAsync(badDebt.ARAPId.Value);
        if (arap == null)
        {
            return OperationResult.Fail("关联的应收/应付单不存在");
        }

        // 校验收付类型一致性
        if (arap.ReceivePaymentType != badDebt.ReceivePaymentType)
        {
            return OperationResult.Fail("收付类型与关联的应收/应付单不一致");
        }

        // 校验金额不超过可用金额
        var availableAmount = await GetAvailableBadDebtAmountAsync(badDebt.ARAPId.Value);
        if (badDebt.BadDebtAmount > availableAmount)
        {
            return OperationResult.Fail($"坏账金额({badDebt.BadDebtAmount})不能超过可用金额({availableAmount})");
        }
    }

    return OperationResult.Success();
}
```

### 3.2 GetAvailableBadDebtAmountAsync - 获取可申请金额

```csharp
/// <summary>
/// 获取应收/应付单的可申请坏账金额
/// = 未核销金额 - 已申请坏账金额（未撤销的）
/// </summary>
/// <param name="arapId">应收/应付单Id</param>
/// <returns>可申请坏账金额</returns>
public async Task<decimal> GetAvailableBadDebtAmountAsync(long arapId)
{
    var arap = await _arapRepo.FindAsync(arapId);
    if (arap == null) return 0;

    // 未核销金额
    decimal unpaidAmount = arap.LocalBalanceAmount;

    // 已申请坏账金额（未撤销的）
    var appliedAmount = await _badDebtRepo.GetSumAsync(
        x => x.ARAPId == arapId &&
             x.BadDebtStatus != (int)BadDebtStatus.已作废 &&
             x.BadDebtStatus != (int)BadDebtStatus.已撤销 &&
             x.isdeleted == false,
        x => x.BadDebtAmount - x.ReversedAmount);

    return unpaidAmount - (appliedAmount ?? 0);
}
```

### 4.1 与应收/应付模块交互

| 交互点 | 方法 | 说明 |
|-------|------|------|

### 4.2 与收付款模块交互

通过核销记录表 `tb_FM_BadDebtWriteOff` 与收付款模块集成：
- `SettlementType = 坏账核销(5)`：正常核销
- `SettlementType = 坏账撤销核销(11)`：撤销时生成的红字核销

### 4.3 与审批流交互

坏账申请单可以接入现有审批流程：
- 提交时触发审批
- 审核通过后执行核销逻辑
- 驳回时状态回退

### 4.4 与损益费用单交互

坏账处理可以关联生成损益费用单：
- 审核通过时，根据坏账原因类型获取对应的损益类型
- 调用 `ProfitLossControllerPartial` 生成费用单
- 将 `ProfitLossType = 坏账准备(10)` 写入费用明细

## 6. 异常处理

### 6.1 业务异常类型

| 异常类型 | 说明 | 处理方式 |
|---------|------|---------|
| `BadDebtNotFoundException` | 坏账申请不存在 | 返回失败结果 |
| `InvalidStatusTransitionException` | 状态转换无效 | 返回失败结果 |
| `AmountExceedException` | 金额超限 | 返回失败结果 |
| `ARAPNotFoundException` | 应收/应付单不存在 | 返回失败结果 |
| `AlreadyReversedException` | 已撤销重复操作 | 返回失败结果 |

### 6.2 事务处理原则

1. **审核操作必须使用事务**：核销余额、创建核销记录、更新状态必须在同一个事务中
2. **撤销操作必须使用事务**：恢复余额、创建红字核销、更新状态必须在同一个事务中
3. **其他操作可选择使用事务**：根据业务复杂度决定
