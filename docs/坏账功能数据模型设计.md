## 1. 数据表设计

### 1.1 坏账申请表 (tb_FM_BadDebt)

```sql
-- =============================================
-- 坏账申请表
-- 用于记录坏账申请的完整信息
-- 支持坏账撤销（通过BadDebtStatus状态控制）
-- =============================================
IF NOT EXISTS (SELECT * FROM sys.objects WHERE object_id = OBJECT_ID(N'[dbo].[tb_FM_BadDebt]') AND type in (N'U'))
BEGIN
    CREATE TABLE [dbo].[tb_FM_BadDebt](
        [BadDebtId]              BIGINT           NOT NULL,  -- 坏账Id
        [BadDebtNo]              VARCHAR(30)       NOT NULL,  -- 坏账单号
        [ARAPId]                 BIGINT           NULL,      -- 关联应收付款单Id
        [ARAPNo]                 VARCHAR(30)       NULL,      -- 关联应收付款单号
        [SourceBillId]          BIGINT           NULL,      -- 来源单据Id
        [SourceBillNo]          VARCHAR(30)       NULL,      -- 来源单据号
        [SourceBizType]         INT              NULL,      -- 来源业务类型
        [ReceivePaymentType]    INT              NOT NULL,  -- 收付类型(1:应收,2:应付)

        -- 往来信息
        [CustomerVendor_ID]      BIGINT           NOT NULL,  -- 往来单位
        [Currency_ID]           BIGINT           NOT NULL,  -- 币别
        [ExchangeRate]          DECIMAL(18,4)    NOT NULL DEFAULT 1,  -- 汇率

        -- 金额信息
        [OriginalAmount]         DECIMAL(18,4)    NOT NULL DEFAULT 0,  -- 原单据金额
        [UnpaidAmount]          DECIMAL(18,4)    NOT NULL DEFAULT 0,  -- 申请时未核销金额
        [BadDebtAmount]         DECIMAL(18,4)    NOT NULL DEFAULT 0,  -- 申请坏账金额(本币)
        [BadDebtForeignAmount]  DECIMAL(18,4)    NULL,      -- 坏账外币金额
        [ReversedAmount]        DECIMAL(18,4)    NOT NULL DEFAULT 0,  -- 已撤销金额(用于部分撤销)
        [RemainingAmount]       DECIMAL(18,4)    NOT NULL DEFAULT 0,  -- 剩余坏账金额

        -- 坏账原因
        [BadDebtReasonType]     INT              NOT NULL,  -- 坏账原因类型
        [BadDebtReason]         NVARCHAR(500)    NULL,      -- 坏账原因说明

        -- 关联单据
        [RelatedDocumentNo]     VARCHAR(100)     NULL,      -- 相关单据号

        -- 组织信息
        [DepartmentID]          BIGINT           NULL,      -- 部门
        [Employee_ID]           BIGINT           NOT NULL,  -- 经办人

        -- 状态信息
        [DataStatus]           INT              NULL,      -- 数据状态(继承状态管理系统)
        [BadDebtStatus]        INT              NOT NULL DEFAULT 0,  -- 坏账处理状态
        [ProcessedAt]          DATETIME         NULL,      -- 处理时间

        -- 审批信息
        [Approver_by]          BIGINT           NULL,      -- 审批人
        [Approver_at]          DATETIME         NULL,      -- 审批时间
        [ApprovalOpinions]      VARCHAR(255)     NULL,      -- 审批意见
        [ApprovalStatus]        TINYINT          NULL,      -- 审批状态
        [ApprovalResults]       BIT              NULL,      -- 审批结果

        -- 撤销信息
        [IsReversed]           BIT              NOT NULL DEFAULT 0,  -- 是否已撤销
        [ReversedAt]           DATETIME         NULL,      -- 撤销时间
        [ReversedBy]           BIGINT           NULL,      -- 撤销人
        [ReversedReason]       NVARCHAR(500)    NULL,      -- 撤销原因

        -- 核销信息
        [IsWriteOff]           BIT              NOT NULL DEFAULT 0,  -- 是否已核销
        [WriteOffAt]           DATETIME         NULL,      -- 核销时间

        -- 通用字段
        [Remark]               NVARCHAR(500)    NULL,      -- 备注
        [Created_at]           DATETIME         NULL,      -- 创建时间
        [Created_by]           BIGINT           NULL,      -- 创建人
        [Modified_at]           DATETIME         NULL,      -- 修改时间
        [Modified_by]          BIGINT           NULL,      -- 修改人
        [isdeleted]            BIT              NOT NULL DEFAULT 0,  -- 逻辑删除
        [PrintStatus]           INT              NOT NULL DEFAULT 0,  -- 打印状态

        CONSTRAINT [PK_tb_FM_BadDebt] PRIMARY KEY CLUSTERED ([BadDebtId] ASC),
        CONSTRAINT [FK_tb_FM_BadDebt_tb_FM_ReceivablePayable] FOREIGN KEY ([ARAPId])
            REFERENCES [dbo].[tb_FM_ReceivablePayable] ([ARAPId]),
        CONSTRAINT [FK_tb_FM_BadDebt_tb_CustomerVendor] FOREIGN KEY ([CustomerVendor_ID])
            REFERENCES [dbo].[tb_CustomerVendor] ([CustomerVendor_ID])
    )
END
GO

CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebt_ARAPId] ON [dbo].[tb_FM_BadDebt]([ARAPId])
CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebt_BadDebtNo] ON [dbo].[tb_FM_BadDebt]([BadDebtNo])
CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebt_CustomerVendor_ID] ON [dbo].[tb_FM_BadDebt]([CustomerVendor_ID])
CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebt_BadDebtStatus] ON [dbo].[tb_FM_BadDebt]([BadDebtStatus])
CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebt_IsReversed] ON [dbo].[tb_FM_BadDebt]([IsReversed])
GO
```

### 1.2 坏账核销记录表 (tb_FM_BadDebtWriteOff)

```sql
-- =============================================
-- 坏账核销记录表
-- 记录坏账审核通过后的核销明细
-- 支持红字核销（撤销时生成）
-- =============================================
IF NOT EXISTS (SELECT * FROM sys.objects WHERE object_id = OBJECT_ID(N'[dbo].[tb_FM_BadDebtWriteOff]') AND type in (N'U'))
BEGIN
    CREATE TABLE [dbo].[tb_FM_BadDebtWriteOff](
        [WriteOffId]               BIGINT         NOT NULL,  -- 核销Id
        [WriteOffNo]               VARCHAR(30)    NOT NULL,  -- 核销单号

        -- 关联信息
        [BadDebtId]                BIGINT         NOT NULL,  -- 坏账申请Id
        [BadDebtNo]                VARCHAR(30)    NOT NULL,  -- 坏账申请单号
        [ARAPId]                   BIGINT         NOT NULL,  -- 应收付款单Id
        [ARAPNo]                   VARCHAR(30)    NOT NULL,  -- 应收付款单号
        [ReceivePaymentType]       INT           NOT NULL,  -- 收付类型

        -- 往来信息
        [CustomerVendor_ID]        BIGINT        NOT NULL,  -- 往来单位
        [Currency_ID]              BIGINT        NOT NULL,  -- 币别
        [ExchangeRate]             DECIMAL(18,4) NOT NULL DEFAULT 1,  -- 汇率

        -- 金额信息
        [OriginalLocalBalance]     DECIMAL(18,4) NOT NULL DEFAULT 0,  -- 核销前余额
        [WriteOffLocalAmount]     DECIMAL(18,4) NOT NULL DEFAULT 0,  -- 核销本币金额(正值表示核销，负值表示撤销)
        [WriteOffForeignAmount]   DECIMAL(18,4) NULL,      -- 核销外币金额
        [AfterWriteOffLocalBalance] DECIMAL(18,4) NOT NULL DEFAULT 0,  -- 核销后余额

        -- 核销类型
        [SettlementType]           INT           NOT NULL DEFAULT 5,  -- 核销类型(5:坏账核销)
        [IsReversedWriteOff]      BIT           NOT NULL DEFAULT 0,  -- 是否是撤销核销(红字)

        -- 时间信息
        [WriteOffDate]            DATETIME      NOT NULL,  -- 核销日期

        -- 组织信息
        [Employee_ID]             BIGINT        NOT NULL,  -- 经办人

        -- 备注
        [Remark]                  NVARCHAR(500) NULL,      -- 备注

        -- 通用字段
        [Created_at]              DATETIME      NULL,      -- 创建时间
        [Created_by]              BIGINT        NULL       -- 创建人

        CONSTRAINT [PK_tb_FM_BadDebtWriteOff] PRIMARY KEY CLUSTERED ([WriteOffId] ASC),
        CONSTRAINT [FK_tb_FM_BadDebtWriteOff_tb_FM_BadDebt] FOREIGN KEY ([BadDebtId])
            REFERENCES [dbo].[tb_FM_BadDebt] ([BadDebtId]),
        CONSTRAINT [FK_tb_FM_BadDebtWriteOff_tb_FM_ReceivablePayable] FOREIGN KEY ([ARAPId])
            REFERENCES [dbo].[tb_FM_ReceivablePayable] ([ARAPId])
    )
END
GO

CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebtWriteOff_BadDebtId] ON [dbo].[tb_FM_BadDebtWriteOff]([BadDebtId])
CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebtWriteOff_ARAPId] ON [dbo].[tb_FM_BadDebtWriteOff]([ARAPId])
CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebtWriteOff_WriteOffDate] ON [dbo].[tb_FM_BadDebtWriteOff]([WriteOffDate])
CREATE NONCLUSTERED INDEX [IX_tb_FM_BadDebtWriteOff_IsReversedWriteOff] ON [dbo].[tb_FM_BadDebtWriteOff]([IsReversedWriteOff])
GO
```
