# 日志系统使用示例

## 1. 审计日志使用示例

### 1.1 在控制器中使用

```csharp
public class ProductController
{
    private readonly IAuditLogService _auditLogService;
    private readonly ILogger<ProductController> _logger;
    private readonly IUnitOfWorkManage _unitOfWorkManage;

    public ProductController(
        IAuditLogService auditLogService,
        ILogger<ProductController> logger,
        IUnitOfWorkManage unitOfWorkManage)
    {
        _auditLogService = auditLogService;
        _logger = logger;
        _unitOfWorkManage = unitOfWorkManage;
    }

    /// <summary>
    /// 添加产品
    /// </summary>
    public async Task<ReturnResults> AddProduct(tb_Product product)
    {
        try
        {
            // 验证（无事务）
            if (product == null)
            {
                _logger.LogWarning("添加产品失败：产品为空");
                return ReturnResults.Error("产品不能为空");
            }

            // 开启事务
            _unitOfWorkManage.BeginTran();
            try
            {
                // 业务逻辑
                await _productService.AddAsync(product);

                // 记录审计日志（异步，不阻塞业务）
                await _auditLogService.LogAsync("添加产品", product, "添加新产品");

                _unitOfWorkManage.CommitTran();
                return ReturnResults.Success("产品添加成功");
            }
            catch (Exception ex)
            {
                _unitOfWorkManage.RollbackTran();
                _logger.LogError(ex, "添加产品失败 | ProductName:{ProductName}", product.ProductName);
                return ReturnResults.Error($"添加产品失败: {ex.Message}");
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "添加产品操作异常");
            return ReturnResults.Error("系统异常");
        }
    }

    /// <summary>
    /// 更新产品
    /// </summary>
    public async Task<ReturnResults> UpdateProduct(tb_Product product)
    {
        try
        {
            // 验证（无事务）
            if (product == null || product.ID <= 0)
            {
                return ReturnResults.Error("产品信息无效");
            }

            _unitOfWorkManage.BeginTran();
            try
            {
                await _productService.UpdateAsync(product);
                await _auditLogService.LogAsync("更新产品", product, $"更新产品信息");

                _unitOfWorkManage.CommitTran();
                return ReturnResults.Success("产品更新成功");
            }
            catch (Exception ex)
            {
                _unitOfWorkManage.RollbackTran();
                _logger.LogError(ex, "更新产品失败 | ProductID:{ProductID}", product.ID);
                return ReturnResults.Error($"更新产品失败: {ex.Message}");
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "更新产品操作异常");
            return ReturnResults.Error("系统异常");
        }
    }

    /// <summary>
    /// 删除产品
    /// </summary>
    public async Task<ReturnResults> DeleteProduct(long productId)
    {
        try
        {
            // 验证（无事务）
            if (productId <= 0)
            {
                return ReturnResults.Error("产品ID无效");
            }

            var product = await _productService.GetByIdAsync(productId);
            if (product == null)
            {
                _logger.LogWarning("删除产品失败：产品不存在 | ProductID:{ProductID}", productId);
                return ReturnResults.Error("产品不存在");
            }

            _unitOfWorkManage.BeginTran();
            try
            {
                await _productService.DeleteAsync(productId);
                await _auditLogService.LogAsync("删除产品", product, $"删除产品: {product.ProductName}");

                _unitOfWorkManage.CommitTran();
                return ReturnResults.Success("产品删除成功");
            }
            catch (Exception ex)
            {
                _unitOfWorkManage.RollbackTran();
                _logger.LogError(ex, "删除产品失败 | ProductID:{ProductID}", productId);
                return ReturnResults.Error($"删除产品失败: {ex.Message}");
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "删除产品操作异常");
            return ReturnResults.Error("系统异常");
        }
    }

    /// <summary>
    /// 记录简单审计日志（无实体）
    /// </summary>
    public async Task<ReturnResults> BatchDeleteProducts(List<long> productIds)
    {
        try
        {
            if (productIds == null || productIds.Count == 0)
            {
                return ReturnResults.Error("产品ID列表为空");
            }

            _unitOfWorkManage.BeginTran();
            try
            {
                foreach (var id in productIds)
                {
                    await _productService.DeleteAsync(id);
                }

                // 记录简单审计日志
                await _auditLogService.LogAsync("批量删除产品", $"批量删除了 {productIds.Count} 个产品");

                _unitOfWorkManage.CommitTran();
                return ReturnResults.Success($"成功删除 {productIds.Count} 个产品");
            }
            catch (Exception ex)
            {
                _unitOfWorkManage.RollbackTran();
                _logger.LogError(ex, "批量删除产品失败");
                return ReturnResults.Error($"批量删除产品失败: {ex.Message}");
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "批量删除产品操作异常");
            return ReturnResults.Error("系统异常");
        }
    }
}
```

```csharp
        {
```

```csharp
            var products = await _repository.GetAllAsync();

    /// <summary>
    /// 导入产品数据
    /// </summary>
    public async Task ImportProducts(List<tb_Product> products)
    {
        try
        {
            _logger.LogInformation("开始导入产品数据 | 数量:{Count}", products.Count);

            var successCount = 0;
            var failedCount = 0;

            foreach (var product in products)
            {
                try
                {
                    await _repository.AddAsync(product);
                    successCount++;
                    _logger.LogDebug("产品导入成功 | ProductID:{ProductID}", product.ID);
                }
                catch (Exception ex)
                {
                    failedCount++;
                    _logger.LogWarning(ex, "产品导入失败 | ProductName:{ProductName}", product.ProductName);
                }
            }

            _logger.LogInformation(
                "产品导入完成 | 成功:{SuccessCount} 失败:{FailedCount}", 
                successCount, failedCount);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "导入产品数据异常");
            throw;
        }
    }

    /// 性能监控

        {
```

```csharp
public class InventoryService
{
    private readonly ILogger<InventoryService> _logger;

    public InventoryService(ILogger<InventoryService> logger)
    {
        _logger = logger;
    }

    /// <summary>
    /// 库存操作示例
    /// </summary>
    public async Task AdjustInventory(long productId, int quantity, string reason)
    {
        try
        {
            // Trace - 最详细的诊断信息（默认不启用）
            _logger.LogTrace("库存调整开始 | ProductID:{ProductID} | Quantity:{Quantity} | Reason:{Reason}", 
                productId, quantity, reason);

            var inventory = await _repository.GetByProductIdAsync(productId);

            // Debug - 调试信息
            _logger.LogDebug("当前库存 | ProductID:{ProductID} | CurrentQuantity:{CurrentQuantity}", 
                productId, inventory?.Quantity ?? 0);

            // Information - 一般信息流（生产环境推荐）
            _logger.LogInformation("库存调整 | ProductID:{ProductID} | Adjustment:{Adjustment} | Reason:{Reason}", 
                productId, quantity, reason);

            // Warning - 警告信息（需要关注但不是错误）
            if (inventory?.Quantity < 0)
            {
                _logger.LogWarning("库存不足 | ProductID:{ProductID} | CurrentQuantity:{CurrentQuantity}", 
                    productId, inventory.Quantity);
            }

            // Error - 错误信息
            if (inventory == null)
            {
                _logger.LogError("库存记录不存在 | ProductID:{ProductID}", productId);
                throw new Exception("库存记录不存在");
            }

            await _repository.UpdateQuantityAsync(productId, quantity);

            _logger.LogInformation("库存调整成功 | ProductID:{ProductID} | NewQuantity:{NewQuantity}", 
                productId, inventory.Quantity + quantity);
        }
        catch (Exception ex)
        {
            // Error - 记录异常
            _logger.LogError(ex, 
                "库存调整失败 | ProductID:{ProductID} | Quantity:{Quantity} | Reason:{Reason}", 
                productId, quantity, reason);
            throw;
        }
    }
}
```

## 4. 最佳实践

### 4.1 事务与日志的配合

```csharp
// ✅ 正确：先验证，再开启事务
public async Task<ReturnResults> Operation(Entity entity)
{
    // 1. 只读验证（无事务）
    var validation = await ValidateAsync(entity);
    if (!validation.IsValid)
    {
        _logger.LogWarning("验证失败 | Errors:{Errors}", string.Join(",", validation.Errors));
        return ReturnResults.Error(validation.Errors);
    }

    // 2. 开启事务
    _unitOfWorkManage.BeginTran();
    try
    {
        // 3. 业务逻辑
        await CoreBusinessLogicAsync(entity);

        // 4. 记录审计日志（异步，不阻塞）
        await _auditLogService.LogAsync("操作", entity);

        // 5. 提交事务
        _unitOfWorkManage.CommitTran();
        return ReturnResults.Success();
    }
    catch (Exception ex)
    {
        // 6. 必须回滚
        _unitOfWorkManage.RollbackTran();
        
        // 7. 必须记录日志
        _logger.LogError(ex, "业务操作失败");
        
        return ReturnResults.Error(ex.Message);
    }
}

// ❌ 错误：过早开启事务
public async Task<ReturnResults> BadOperation(Entity entity)
{
    _unitOfWorkManage.BeginTran();  // 过早开启！
    try
    {
        var validation = await ValidateAsync(entity);  // 验证也持有事务
        if (!validation.IsValid)
        {
            _unitOfWorkManage.RollbackTran();  // 不必要的回滚
            return ReturnResults.Error(validation.Errors);
        }
        
        await CoreBusinessLogicAsync(entity);
        await _auditLogService.LogAsync("操作", entity);
        _unitOfWorkManage.CommitTran();
        return ReturnResults.Success();
    }
    catch (Exception ex)
    {
        _unitOfWorkManage.RollbackTran();
        _logger.LogError(ex);
        return ReturnResults.Error(ex.Message);
    }
}
```

## 5. 总结

### 5.1 日志系统特点

1. **审计日志优化**
   - 异步队列批量写入
   - 队列限流保护
   - 自动错误重试
   - 性能监控指标

2. **普通日志优化**
   - 支持缓冲模式
   - 自动区分重要/高频日志
   - 上下文属性自动设置

3. **无缝集成**
   - 保持原有API接口
   - 通过DI容器注入
   - 配置灵活可调

### 5.2 使用要点

1. 审计日志调用 `IAuditLogService.LogAsync`
2. 普通日志使用 `ILogger<T>`
3. 事务与日志正确配合
4. 使用结构化日志格式
5. 异常日志必须包含上下文信息
