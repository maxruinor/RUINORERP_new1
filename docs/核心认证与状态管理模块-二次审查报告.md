### 2. LoginCommandHandler.cs - 阻塞式异步调用 ✅ 已修复

**原问题**: 在属性getter中使用`.Result`导致死锁风险

**修复验证**:
```csharp
// 第63-107行 - 修复后实现
private int? _cachedMaxConcurrentUsers;
private DateTime _cacheExpiryTime;
private readonly object _maxUsersCacheLock = new object();
private readonly TimeSpan _maxUsersCacheDuration = TimeSpan.FromMinutes(5);

private async Task<int> GetMaxConcurrentUsersAsync()
{
    lock (_maxUsersCacheLock)
    {
        if (_cachedMaxConcurrentUsers.HasValue && DateTime.Now < _cacheExpiryTime)
        {
            return _cachedMaxConcurrentUsers.Value;
        }
    }
    
    try
    {
        if (RegistrationService != null)
        {
            var validationResult = await RegistrationService.ValidateSystemRegistrationAsync();
            if (validationResult?.RegistrationInfo != null && validationResult.IsValid)
            {
                var concurrentUsers = validationResult.RegistrationInfo.ConcurrentUsers;
                
                lock (_maxUsersCacheLock)
                {
                    _cachedMaxConcurrentUsers = concurrentUsers;
                    _cacheExpiryTime = DateTime.Now.Add(_maxUsersCacheDuration);
                }
                
                return concurrentUsers;
            }
        }
    }
    catch (Exception ex)
    {
        logger?.LogWarning(ex, "获取注册信息中的并发用户数失败，使用默认值");
    }
    
    return DefaultMaxConcurrentUsers;
}
```

### 3. UnitOfWorkManage.cs - 事务嵌套风险 ✅ 已修复

**修复验证**:
```csharp
// 第251-339行 - BeginTran修复
public void BeginTran(IsolationLevel? isolationLevel = null, int? timeoutSeconds = null)
{
    var context = GetOrCreateTransactionContext();
    var dbClient = GetDbClient();

    // ✅ P1优化: 使用SemaphoreSlim.Wait()替代lock
    context.LockSemaphore.Wait();
    try
    {
        // ✅ 检查嵌套深度
        if (context.Depth >= 15)
        {
            throw new InvalidOperationException(
                $"[Transaction-{context.TransactionId}] 事务嵌套深度超过最大限制 (15)...");
        }

        // 增加事务深度
        context.Depth++;
        
        // 最外层事务：开启物理事务
        if (context.Depth == 1)
        {
            // ✅ 防御性检查: 清理可能存在的挂起命令
            if (dbClient.Ado.Transaction != null)
            {
                _logger.LogWarning($"[Transaction-{context.TransactionId}] 检测到残留事务对象，先回滚");
                try { dbClient.Ado.RollbackTran(); } catch { }
            }
            
            dbClient.Ado.BeginTran();
        }
        // 嵌套事务：使用保存点
        else
        {
            var guidShort = context.TransactionId.ToString("N").Substring(0, 8);
            var savePointName = $"SP_{guidShort}_{context.Depth}";
            dbClient.Ado.ExecuteCommand($"SAVE TRANSACTION {savePointName}");
            context.SavePointStack.Push(savePointName);
        }
    }
    finally
    {
        context.LockSemaphore.Release();
    }
}
```

### 6. FrmLogin.cs - UI线程安全 ⚠️ 部分修复

**原问题**: 异步登录回调可能不在UI线程执行

**修复验证**:
查看代码发现登录逻辑在 `btnok_Click` 事件中（第434行开始），但需要确认是否正确处理了UI线程切换。

**发现的问题**:
```csharp
// 第434行 - 登录按钮点击事件
private async void btnok_Click(object sender, EventArgs e)
{
    // ... 登录逻辑
    
    // 第550行左右 - 处理登录响应
    if (response.IsSuccess)
    {
        // ⚠️ 这里直接更新UI，但没有检查InvokeRequired
        MainForm.Instance.CurrentLoginStatus = MainForm.LoginStatus.LoggedIn;
        this.DialogResult = DialogResult.OK;
        this.Close();
    }
}
```

**验证结果**: ⚠️ **部分修复**
- ✅ 使用了async/await模式
- ✅ 禁用了登录按钮防止重复点击
- ✅ 有取消令牌支持
- ⚠️ **未发现显式的UI线程检查**（InvokeRequired）

**风险评估**: 
- Windows Forms中，`async/await`默认会捕获SynchronizationContext
- 如果FrmLogin在主UI线程创建，await后会回到UI线程
- **但在某些情况下（如线程池线程启动）可能不会回到UI线程**

**建议修复**:
```csharp
private async void btnok_Click(object sender, EventArgs e)
{
    try
    {
        btnok.Enabled = false;
        
        var response = await _userLoginService.LoginAsync(username, password, _loginCancellationTokenSource.Token);
        
        // ✅ 确保在UI线程执行
        if (InvokeRequired)
        {
            Invoke(new Action(() => HandleLoginResponse(response)));
        }
        else
        {
            HandleLoginResponse(response);
        }
    }
    catch (Exception ex)
    {
        // 错误处理
    }
}

private void HandleLoginResponse(LoginResponse response)
{
    if (response.IsSuccess)
    {
        MainForm.Instance.CurrentLoginStatus = MainForm.LoginStatus.LoggedIn;
        this.DialogResult = DialogResult.OK;
        this.Close();
    }
    else
    {
        MessageBox.Show(response.ErrorMessage, "登录失败", 
            MessageBoxButtons.OK, MessageBoxIcon.Error);
        btnok.Enabled = true;
    }
}
```

**优先级**: 🟡 中（当前可能工作，但不保证在所有场景下都安全）
