### 2. LoginCommandHandler.cs - 阻塞式异步调用

**位置**: 第64-88行 `MaxConcurrentUsers` 属性  
**严重程度**: 🔴 高  
**影响范围**: 可能导致死锁，违反async/await最佳实践

#### 问题描述
```csharp
private int MaxConcurrentUsers
{
    get
    {
        try
        {
            if (RegistrationService != null)
            {
                // ⚠️ 在同步属性中使用 .Result 会导致死锁风险
                var validationResult = RegistrationService.ValidateSystemRegistrationAsync().Result;
                if (validationResult?.RegistrationInfo != null && validationResult.IsValid)
                {
                    return validationResult.RegistrationInfo.ConcurrentUsers;
                }
            }
        }
        catch (Exception ex)
        {
            logger?.LogWarning(ex, "获取注册信息中的并发用户数失败，使用默认值");
        }
        
        return DefaultMaxConcurrentUsers;
    }
}
```

**问题分析**:
- 在属性getter中调用`.Result`会阻塞线程
- 如果`ValidateSystemRegistrationAsync`内部也等待其他任务，可能导致死锁
- 每次访问都调用异步方法，性能差

#### 修复建议
```csharp
// 方案1: 添加缓存机制
private int? _cachedMaxConcurrentUsers;
private DateTime _cacheExpiryTime;
private readonly object _cacheLock = new object();

private async Task<int> GetMaxConcurrentUsersAsync()
{
    lock (_cacheLock)
    {
        if (_cachedMaxConcurrentUsers.HasValue && DateTime.Now < _cacheExpiryTime)
            return _cachedMaxConcurrentUsers.Value;
    }
    
    try
    {
        if (RegistrationService != null)
        {
            var validationResult = await RegistrationService.ValidateSystemRegistrationAsync();
            if (validationResult?.RegistrationInfo != null && validationResult.IsValid)
            {
                var count = validationResult.RegistrationInfo.ConcurrentUsers;
                
                lock (_cacheLock)
                {
                    _cachedMaxConcurrentUsers = count;
                    _cacheExpiryTime = DateTime.Now.AddMinutes(5); // 缓存5分钟
                }
                
                return count;
            }
        }
    }
    catch (Exception ex)
    {
        logger?.LogWarning(ex, "获取注册信息中的并发用户数失败，使用默认值");
    }
    
    return DefaultMaxConcurrentUsers;
}

// 在需要的地方异步调用
var maxUsers = await GetMaxConcurrentUsersAsync();
```

### 3. UnitOfWorkManage.cs - 事务嵌套风险

**位置**: `BeginTransaction()` 方法  
**严重程度**: 🔴 高  
**影响范围**: 可能导致事务异常或资源泄漏

#### 问题描述
```csharp
public void BeginTransaction()
{
    // ⚠️ 未检查是否已有活动事务
    _transaction = _db.Ado.BeginTran();
}
```

**问题分析**:
- 如果已经存在活动事务，再次调用会导致异常
- 未对事务状态进行跟踪
- 可能导致资源泄漏

#### 修复建议
```csharp
private bool _isInTransaction = false;

public void BeginTransaction()
{
    if (_isInTransaction)
    {
        _logger?.LogWarning("事务已存在，跳过重复开启");
        return;
    }
    
    try
    {
        _transaction = _db.Ado.BeginTran();
        _isInTransaction = true;
        _logger?.LogDebug("事务已开启");
    }
    catch (Exception ex)
    {
        _logger?.LogError(ex, "开启事务失败");
        throw;
    }
}

public void CommitTransaction()
{
    if (!_isInTransaction)
    {
        _logger?.LogWarning("无活动事务，跳过提交");
        return;
    }
    
    try
    {
        _db.Ado.CommitTran();
        _isInTransaction = false;
        _logger?.LogDebug("事务已提交");
    }
    catch (Exception ex)
    {
        _logger?.LogError(ex, "提交事务失败");
        RollbackTransaction();
        throw;
    }
}

public void RollbackTransaction()
{
    if (!_isInTransaction)
    {
        _logger?.LogWarning("无活动事务，跳过回滚");
        return;
    }
    
    try
    {
        _db.Ado.RollbackTran();
        _logger?.LogWarning("事务已回滚");
    }
    catch (Exception ex)
    {
        _logger?.LogError(ex, "回滚事务失败");
    }
    finally
    {
        _isInTransaction = false;
    }
}
```

**位置**: 第83-96行 `SetSubmitModifyRuleMode`  
**严重程度**: 🟡 中  
**影响范围**: 多线程环境下规则初始化可能出错

#### 问题描述
```csharp

{
    if (submitModifyRuleMode != mode)
    {
        submitModifyRuleMode = mode;
        
        if (_isInitialized)
```

- 其他线程可能在规则重置后、初始化前访问规则字典

```csharp
public void SetSubmitModifyRuleMode(SubmitModifyRuleMode mode)
{

    {
        lock (_initLock)
        {
            submitModifyRuleMode = mode;
            
            if (_isInitialized)
            {
                _logger?.LogInformation($"切换提交修改模式: {mode}");
                ResetAllRules();
                InitializeAllRules();
                _logger?.LogInformation("规则重新初始化完成");
            }
        }
    }
}

// 同样需要保护 InitializeAllRules
public void InitializeAllRules()
{
    lock (_initLock)
    {
        if (_isInitialized)
            return;

        _stateTransitionRules.Clear();
        _uiButtonRules.Clear();
        _actionPermissionRules.Clear();

        InitializeStateTransitionRules();
        InitializeUIButtonRules();
        InitializeActionPermissionRules();

        _isInitialized = true;
        _logger?.LogInformation("全局状态规则初始化完成");
    }
}
```

### 5. SessionService.cs - 会话清理策略不明确

**位置**: 需要查看完整实现  
**严重程度**: 🟡 中  
**影响范围**: 可能导致内存泄漏

#### 问题描述
- 过期会话清理时机不明确
- 未看到定期清理机制
- 长时间运行可能导致内存占用持续增长

#### 修复建议
```csharp
public class SessionService : ISessionService
{
    private readonly ConcurrentDictionary<string, SessionInfo> _sessions = new();
    private readonly Timer _cleanupTimer;
    private readonly TimeSpan _sessionTimeout = TimeSpan.FromMinutes(30);
    private readonly ILogger<SessionService> _logger;

    public SessionService(ILogger<SessionService> logger)
    {
        _logger = logger;
        
        // 启动定期清理任务
        _cleanupTimer = new Timer(
            _ => CleanupExpiredSessions(),
            null,
            TimeSpan.FromMinutes(10),  // 首次延迟10分钟
            TimeSpan.FromMinutes(10)   // 每10分钟执行一次
        );
        
        _logger?.LogInformation("会话清理定时器已启动");
    }

    private void CleanupExpiredSessions()
    {
        try
        {
            var now = DateTime.Now;
            var expiredKeys = new List<string>();

            foreach (var kvp in _sessions)
            {
                var session = kvp.Value;
                
                // 检查会话是否过期
                if (session.LastActivityTime < now - _sessionTimeout)
                {
                    expiredKeys.Add(kvp.Key);
                }
            }

            // 移除过期会话
            foreach (var key in expiredKeys)
            {
                if (_sessions.TryRemove(key, out var removedSession))
                {
                    _logger?.LogInformation(
                        $"清理过期会话: SessionId={key}, UserName={removedSession.UserName}, LastActivity={removedSession.LastActivityTime}");
                }
            }

            if (expiredKeys.Count > 0)
            {
                _logger?.LogDebug($"本次清理了 {expiredKeys.Count} 个过期会话");
            }
        }
        catch (Exception ex)
        {
            _logger?.LogError(ex, "清理过期会话时发生错误");
        }
    }

    public void Dispose()
    {
        _cleanupTimer?.Dispose();
    }
}
```

### 1. 并发用户数查询优化

```csharp
private ConcurrentDictionary<long, HashSet<string>> _userSessions = new();

public void AddSession(string sessionId, long userId)
{
    _userSessions.AddOrUpdate(
        userId,
        new HashSet<string> { sessionId },
        (key, existing) =>
        {
            existing.Add(sessionId);
            return existing;
        }
    );
}

public void RemoveSession(string sessionId, long userId)
{
    if (_userSessions.TryGetValue(userId, out var sessions))
    {
        sessions.Remove(sessionId);
        if (sessions.Count == 0)
        {
            _userSessions.TryRemove(userId, out _);
        }
    }
}

public int GetActiveUserCount()
{
    return _userSessions.Count;
}
```

```csharp
        if (_stateTransitionRules.TryGetValue(entityType, out var transitions) &&
            transitions.TryGetValue(currentState, out var allowedStates))
        {
            return allowedStates;
        }
        
        return new List<object>();
    });
}
```
