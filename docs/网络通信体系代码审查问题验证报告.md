#### 问题8: GetTimeoutByCommandType效率问题

**当前位置**: `ClientCommunicationService.cs:1344-1379`

**当前实现**:
```csharp
private int GetTimeoutByCommandType(CommandId commandId)
{
    // 缓存相关请求设置较短超时，避免UI阻塞
    if (commandId.Name.Contains("Cache"))
    {
        return 5000; // 缓存请求5秒超时
    }
    
    if (commandId.Name.Contains("Auth") || commandId.Name.Contains("Login"))
    {
        return 8000; // 认证/登录请求8秒超时
    }
    
    if (commandId.Name.Contains("Query") || commandId.Name.Contains("Search"))
    {
        return 10000; // 查询/搜索请求10秒超时
    }
    
    if (commandId.Name.Contains("Save") || commandId.Name.Contains("Update") || commandId.Name.Contains("Delete"))
    {
        return 15000; // 保存/更新/删除请求15秒超时
    }
    
    // 报表相关请求
    if (commandId.Name.Contains("Report") || commandId.Name.Contains("Export"))
    {
        return 30000; // 报表/导出请求30秒超时
    }
    
    // 默认超时时间
    return 20000; // 默认20秒超时
}
```

**问题分析**:
- 🟡 **效率低**: 每次都要执行多次字符串Contains操作
- 🟡 **维护困难**: 新增命令类型需要修改多处
- 🟢 **实际影响小**: 该方法调用频率不高

**优化建议**:
```csharp
private static readonly Dictionary<string, int> CommandTimeoutMap = new Dictionary<string, int>
{
    { "Cache", 5000 },
    { "Auth", 8000 },
    { "Login", 8000 },
    { "Query", 10000 },
    { "Search", 10000 },
    { "Save", 15000 },
    { "Update", 15000 },
    { "Delete", 15000 },
    { "Report", 30000 },
    { "Export", 30000 }
};

private int GetTimeoutByCommandType(CommandId commandId)
{
    // 精确匹配优先
    if (CommandTimeoutMap.TryGetValue(commandId.Name, out int timeout))
        return timeout;
    
    // 模糊匹配（仅在必要时）
    foreach (var kvp in CommandTimeoutMap)
    {
        if (commandId.Name.Contains(kvp.Key))
            return kvp.Value;
    }
    
    return 20000; // 默认20秒
}
```

// 方案1: 使用循环而非递归
private async Task ProcessCommandQueueAsync()
{
    if (!await _queueLock.WaitAsync(TimeSpan.Zero))
        return;

    try
    {
        // 使用循环持续处理，直到队列为空
        while (!_queuedCommands.IsEmpty && !_disposed)
        {
            var commandsToProcess = new List<ClientQueuedCommand>();
            
            while (_queuedCommands.TryDequeue(out var command))
            {
                commandsToProcess.Add(command);
                if (commandsToProcess.Count >= 50)
                    break;
            }

            if (commandsToProcess.Count == 0)
                break;

            // 处理命令...

        }
    }

    }
