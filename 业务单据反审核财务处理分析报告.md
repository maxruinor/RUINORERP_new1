## 1. 采购入库反审核分析

### 位置
文件: `tb_PurEntryControllerPartial.cs`
方法: `AntiApprovalAsync` (第277-795行)

### 财务处理逻辑 (第680-770行)

```csharp
#region 财务反审
AuthorizeController authorizeController = _appContext.GetRequiredService<AuthorizeController>();
if (authorizeController.EnableFinancialModule())
{
    #region 反审 反核销预付
    var ctrpayable = _appContext.GetRequiredService<tb_FM_ReceivablePayableController<tb_FM_ReceivablePayable>>();

    //出库时，全部生成应收，账期的。就加上到期日
    //有付款过的。就去预收中抵扣，不够的金额及状态标识出来
    //如果收款了，则不能反审,预收的可以反审
    var ARAPList = await _appContext.Db.Queryable<tb_FM_ReceivablePayable>()
        .Includes(c => c.tb_FM_ReceivablePayableDetails)
        .Where(c => c.SourceBillId == entity.PurEntryID
            && c.TotalLocalPayableAmount > 0 //正向
            && c.SourceBizType == (int)BizType.采购入库单)
        .ToListAsync();
    
    if (ARAPList != null && ARAPList.Count > 0)
    {
        if (ARAPList.Count == 1)
        {
            var result = await ctrpayable.AntiApplyManualPaymentAllocation(ARAPList[0], ReceivePaymentType.付款, true, false);
        }
        else
        {
            //不会为多行。有错误
            _unitOfWorkManage.RollbackTran();
            rs.ErrorMsg = $"采购入库单{entity.PurEntryNo}有多张应付款单，数据重复，请检查数据正确性后，再操作。";
            rs.Succeeded = false;
            return rs;
        }
    }
}
#endregion
}
#endregion
```

### 存在的问题

❌ **问题1：未检查应付款单的核销状态**
- 当前代码直接调用 `AntiApplyManualPaymentAllocation`
- 没有先检查应付款单是否已被部分或全部核销
- 如果应付款单已经和预付款单核销，反审时应该恢复预付款余额

❌ **问题2：未检查是否存在已审核的付款单**
- 如果已经有实际付款发生，不应该允许反审

❌ **问题3：缺少补偿机制**
- 反审成功后，没有删除应付款单
- 没有删除对应的核销记录

❌ **问题4：未记录审计日志**
- 反审财务数据时缺少审计日志记录

## 3. 应收应付反审核核心逻辑分析

### 位置

### 核心方法：AntiApplyManualPaymentAllocation (第1136行开始)

```csharp
public async Task<bool> AntiApplyManualPaymentAllocation(
    tb_FM_ReceivablePayable payable, 
    ReceivePaymentType paymentType,
    bool AutoAntiApproval, 
    bool UseTransaction = true)
{
    bool result = false;
    try
    {
        decimal remainingLocal = payable.LocalBalanceAmount;
        decimal remainingForeign = payable.ForeignBalanceAmount;
        if (UseTransaction)
        {
            _unitOfWorkManage.BeginTran();
        }

        #region 反写预收付款
        List<tb_FM_ReceivablePayable> arapupdatelist = new List<tb_FM_ReceivablePayable>();
        List<tb_FM_PreReceivedPayment> prePaymentsUpdateList = new List<tb_FM_PreReceivedPayment>();
        
        if (payable.ARAPStatus == (int)ARAPStatus.草稿 || payable.ARAPStatus == (int)ARAPStatus.待审核)
        {
            if (AutoAntiApproval)
            {
                await _appContext.Db.DeleteNav(payable).Include(c => c.tb_FM_ReceivablePayableDetails).ExecuteCommandAsync();
            }
        }
        else
        {
            //根据不同情况处理： 如果没有收款记录，
            #region 如果有核销过，或部分支付
            #region 通过核销记录来处理 如果核销记录中是来自预收付的 则恢复，如果是来自收款单的则无法直接反审出库单。
            
            //反核销  再修改应收付本身金额及状态  以核销记录为标准，因为预收付可以弄一半。核销是整单处理
            var Settlements = await _unitOfWorkManage.GetDbClient().Queryable<tb_FM_PaymentSettlement>()
                .Where(x => (x.TargetBillId == payable.ARAPId || x.SourceBillId == payable.ARAPId))
                .Where(x => x.ReceivePaymentType == payable.ReceivePaymentType)
                .Where(x => x.IsReversed == false && x.isdeleted == false)//正向的
                .ToListAsync();
            
            foreach (var Settlement in Settlements)
            {
                if (paymentType == ReceivePaymentType.收款)
                {
                    //预收转应收时
                    if (Settlement.SourceBizType == (int)BizType.预收款单 && Settlement.TargetBizType == (int)BizType.应收款单)
                    {
                        var prePayment = await _unitOfWorkManage.GetDbClient().Queryable<tb_FM_PreReceivedPayment>()
                            .Where(x => x.PreRPID == Settlement.SourceBillId).SingleAsync();
                        
                        if (prePayment != null)
                        {
                            // 恢复预收款余额
                            prePayment.ForeignBalanceAmount += Settlement.SettledForeignAmount;
                            prePayment.LocalBalanceAmount += Settlement.SettledLocalAmount;
                            prePayment.LocalPaidAmount -= Settlement.SettledLocalAmount;
                            prePayment.ForeignPaidAmount -= Settlement.SettledForeignAmount;
                            
                            // 更新预收款状态
                            decimal tolerance = 0.0001m;
                            if (Math.Abs(prePayment.LocalPaidAmount) <= tolerance)
                            {
                                prePayment.PrePaymentStatus = (int)PrePaymentStatus.待核销;
                            }
                            else if (prePayment.LocalPaidAmount < prePayment.LocalBalanceAmount)
                            {
                                prePayment.PrePaymentStatus = (int)PrePaymentStatus.处理中;
                            }
                            
                            payable.LocalBalanceAmount += Settlement.SettledLocalAmount;
                            payable.ForeignBalanceAmount += Settlement.SettledForeignAmount;
                            payable.ForeignPaidAmount -= Settlement.SettledForeignAmount;
                            payable.LocalPaidAmount -= Settlement.SettledLocalAmount;
                            prePaymentsUpdateList.Add(prePayment);

                            // 标记核销记录为已删除
                            Settlement.isdeleted = true;
                            await _unitOfWorkManage.GetDbClient().Updateable(Settlement).UpdateColumns(it => new { it.isdeleted }).ExecuteCommandAsync();
                        }
                    }
                }
            }
        }
    }
}
```

### 实现的优点

✅ **优点1：有恢复预收付余额的逻辑**
- 通过核销记录找到对应的预收付单
- 恢复预收付的余额和已付金额
- 更新预收付的状态（待核销/处理中）

✅ **优点2：标记核销记录为已删除**
- 通过 `isdeleted` 字段标记核销记录

✅ **优点3：支持自动删除草稿状态的应收应付单**
- 对于草稿或待审核状态的应收应付单，可以直接删除

**问题：**

### 位置

### 核心逻辑

```csharp
public async override Task<ReturnResults<T>> AntiApprovalAsync(T ObjectEntity)
{
    // 只有生效状态的才允许反审，其它不能也不需要，有可能可删除。也可能只能红字
    // 获取当前状态
    var currentStatus = (PrePaymentStatus)Enum.ToObject(
        typeof(PrePaymentStatus),
        entity.GetPropertyValue(statusProperty)
    );

    var ValidateValue = StateManager.ValidateBusinessStatusTransitionAsync(currentStatus, PrePaymentStatus.已生效 as Enum);
    if (!ValidateValue.IsSuccess)
    {
        rmrs.ErrorMsg = $"状态为【{currentStatus.ToString()}】的{((ReceivePaymentType)entity.ReceivePaymentType).ToString()}单不可以反审";
        return rmrs;
    }

    var paymentRecordController = _appContext.GetRequiredService<tb_FM_PaymentRecordController<tb_FM_PaymentRecord>>();
    _unitOfWorkManage.BeginTran();

    var PaymentRecordlist = await _appContext.Db.Queryable<tb_FM_PaymentRecord>()
        .Where(c => c.tb_FM_PaymentRecordDetails.Any(d => d.SourceBilllId == entity.PreRPID))
        .ToListAsync();
    
    if (PaymentRecordlist != null && PaymentRecordlist.Count > 0)
    {
        if ((PaymentRecordlist.Any(c => c.PaymentStatus == (int)PaymentStatus.待审核)
            && PaymentRecordlist.Any(c => c.ApprovalStatus == (int)ApprovalStatus.未审核)))
        {
            _unitOfWorkManage.RollbackTran();
            rmrs.ErrorMsg = $"存在【待审核】的{((ReceivePaymentType)entity.ReceivePaymentType).ToString()}单，请联系上级财务删除后才能反审。";
            rmrs.Succeeded = false;
            return rmrs;
        }

        if ((PaymentRecordlist.Any(c => c.PaymentStatus == (int)PaymentStatus.已支付)
            && PaymentRecordlist.Any(c => c.ApprovalStatus == (int)ApprovalStatus.审核通过)))
        {
            _unitOfWorkManage.RollbackTran();
            rmrs.ErrorMsg = $"存在【已支付】的{((ReceivePaymentType)entity.ReceivePaymentType).ToString()}单，不能反审预款单,请联系上级财务，或作退回处理。";
            rmrs.Succeeded = false;
            return rmrs;
        }
        else
        {
            foreach (var item in PaymentRecordlist)
            {
                await _appContext.Db.DeleteNav<tb_FM_PaymentRecord>(item)
                    .Include(c => c.tb_FM_PaymentRecordDetails)
                    .ExecuteCommandAsync();
            }
        }
    }

    entity.PrePaymentStatus = (int)PrePaymentStatus.待审核;
    entity.ApprovalResults = null;
    entity.ApprovalStatus = (int)ApprovalStatus.未审核;
    BusinessHelper.Instance.ApproverEntity(entity);
    
    await _unitOfWorkManage.GetDbClient().Updateable<tb_FM_PreReceivedPayment>(entity).UpdateColumns(...).ExecuteCommandAsync();
    _unitOfWorkManage.CommitTran();
}
```

### 实现的优点

✅ **优点1：有完整的状态检查**
- 检查是否是已生效状态
- 检查状态转换是否允许

✅ **优点2：检查收款单状态**
- 检查是否存在待审核的收款单
- 检查是否存在已支付的收款单

## 7. 改进建议

### 建议1：统一财务单据状态检查方法

创建公共方法来检查财务单据是否可以反审核：

```csharp
public class FinancialDocumentAntiApprovalValidator
{
    public static (bool CanAntiApproval, string ErrorMessage) ValidateForAntiApproval(
        tb_FM_ReceivablePayable arap,
        List<tb_FM_PaymentSettlement> settlements,
        List<tb_FM_PaymentRecord> paymentRecords,
        IAuthorizeController authorizeController)
    {
        decimal tolerance = authorizeController.GetAmountCalculationTolerance();
        
        // 检查1：应收应付是否已被核销
        if (arap.ARAPStatus == (int)ARAPStatus.部分支付 
            || arap.ARAPStatus == (int)ARAPStatus.全部支付)
        {
            // 计算核销比例
            decimal settlementRatio = arap.TotalLocalPayableAmount > 0 
                ? arap.LocalPaidAmount / arap.TotalLocalPayableAmount 
                : 0;
            
            if (settlementRatio >= 1 - tolerance)
            {
                // 已全额核销，不允许反审
                return (false, 
                    $"应收应付单{arap.ARAPNo}已全额核销（{arap.LocalPaidAmount}/{arap.TotalLocalPayableAmount}），不能反审核。" +
                    $"如需修改，请使用红字冲销机制或联系财务处理。");
            }
        }
        
        // 检查2：是否存在已审核的收款单/付款单
        var paidPaymentRecords = paymentRecords
            .Where(c => c.PaymentStatus == (int)PaymentStatus.已支付 
                && c.ApprovalStatus == (int)ApprovalStatus.审核通过)
            .ToList();
        
        if (paidPaymentRecords.Any())
        {
            string paymentType = ((ReceivePaymentType)arap.ReceivePaymentType).ToString();
            decimal paidAmount = paidPaymentRecords.Sum(c => c.TotalLocalAmount);
            return (false, 
                $"应收应付单{arap.ARAPNo}已通过{paymentType}单核销{paidAmount}元，不能反审核。" +
                $"如需处理，请使用{paymentType}单的红字冲销机制。");
        }
        
        // 检查3：核销记录的来源类型
        var settlementFromPaymentRecord = settlements
            .Any(s => s.SourceBizType == (int)BizType.收款单 
                || s.SourceBizType == (int)BizType.付款单);
```

```csharp
public async Task<bool> AntiApplyManualPaymentAllocation(
    tb_FM_ReceivablePayable payable, 
    ReceivePaymentType paymentType,
    bool AutoAntiApproval, 
    bool UseTransaction = true,
    bool DeletePayable = false)
{
    bool result = false;
    try
    {
        decimal remainingLocal = payable.LocalBalanceAmount;
        decimal remainingForeign = payable.ForeignBalanceAmount;
        if (UseTransaction)
        {
            _unitOfWorkManage.BeginTran();
        }

        List<tb_FM_ReceivablePayable> arapupdatelist = new List<tb_FM_ReceivablePayable>();
        List<tb_FM_PreReceivedPayment> prePaymentsUpdateList = new List<tb_FM_PreReceivedPayment>();
        List<long> settlementIdsToDelete = new List<long>();
        
        if (payable.ARAPStatus == (int)ARAPStatus.草稿 
            || payable.ARAPStatus == (int)ARAPStatus.待审核)
        {
            if (AutoAntiApproval && DeletePayable)
            {
                // 删除应收应付单及其明细
                await _appContext.Db.DeleteNav(payable)
                    .Include(c => c.tb_FM_ReceivablePayableDetails)
                    .ExecuteCommandAsync();
                
                // 删除相关的核销记录
                await _appContext.Db.Queryable<tb_FM_PaymentSettlement>()
                    .Where(x => x.TargetBillId == payable.ARAPId || x.SourceBillId == payable.ARAPId)
                    .Where(x => x.ReceivePaymentType == paymentType)
                    .ExecuteCommandAsync();
                
                result = true;
            }
            else
            {
                result = true;
            }
        }
        else
        {
            var Settlements = await _unitOfWorkManage.GetDbClient().Queryable<tb_FM_PaymentSettlement>()
                .Where(x => (x.TargetBillId == payable.ARAPId || x.SourceBillId == payable.ARAPId))
                .Where(x => x.ReceivePaymentType == paymentType)
                .Where(x => x.IsReversed == false && x.isdeleted == false)
                .ToListAsync();
            
            foreach (var Settlement in Settlements)
            {
                if (paymentType == ReceivePaymentType.收款)
                {
                    if (Settlement.SourceBizType == (int)BizType.预收款单 
                        && Settlement.TargetBizType == (int)BizType.应收款单)
                    {
                        var prePayment = await _unitOfWorkManage.GetDbClient().Queryable<tb_FM_PreReceivedPayment>()
                            .Where(x => x.PreRPID == Settlement.SourceBillId).SingleAsync();
                        
                        if (prePayment != null)
                        {
                            decimal oldLocalBalance = prePayment.LocalBalanceAmount;
                            decimal oldForeignBalance = prePayment.ForeignBalanceAmount;
                            decimal oldLocalPaid = prePayment.LocalPaidAmount;
                            decimal oldForeignPaid = prePayment.ForeignPaidAmount;
                            int oldStatus = prePayment.PrePaymentStatus;
                            
                            prePayment.ForeignBalanceAmount += Settlement.SettledForeignAmount;
                            prePayment.LocalBalanceAmount += Settlement.SettledLocalAmount;
                            prePayment.LocalPaidAmount -= Settlement.SettledLocalAmount;
                            prePayment.ForeignPaidAmount -= Settlement.SettledForeignAmount;
                            
                            decimal tolerance = 0.0001m;
                            if (Math.Abs(prePayment.LocalPaidAmount) <= tolerance)
                            {
                                prePayment.PrePaymentStatus = (int)PrePaymentStatus.待核销;
                            }
                            else if (prePayment.LocalPaidAmount < prePayment.LocalBalanceAmount)
                            {
                                prePayment.PrePaymentStatus = (int)PrePaymentStatus.处理中;
                            }
                            
                            payable.LocalBalanceAmount += Settlement.SettledLocalAmount;
                            payable.ForeignBalanceAmount += Settlement.SettledForeignAmount;
                            payable.ForeignPaidAmount -= Settlement.SettledForeignAmount;
                            payable.LocalPaidAmount -= Settlement.SettledLocalAmount;
                            prePaymentsUpdateList.Add(prePayment);
                            
                            // 记录财务审计日志
                            var fMAuditLog = _appContext.GetRequiredService<FMAuditLogHelper>();
                            string auditMessage = $"预收款单反核销：本币余额从 {oldLocalBalance} 增加到 {prePayment.LocalBalanceAmount}，" +
                                $"外币余额从 {oldForeignBalance} 增加到 {prePayment.ForeignBalanceAmount}，" +
                                $"本币已付金额从 {oldLocalPaid} 减少到 {prePayment.LocalPaidAmount}，" +
                                $"外币已付金额从 {oldForeignPaid} 减少到 {prePayment.ForeignPaidAmount}，" +
                                $"状态从 {oldStatus} 变更为 {prePayment.PrePaymentStatus}。" +
                                $"反核销金额：本币 {Settlement.SettledLocalAmount}，外币 {Settlement.SettledForeignAmount}。";
                            fMAuditLog.CreateAuditLog<tb_FM_PreReceivedPayment>(auditMessage, prePayment);
                            
                            settlementIdsToDelete.Add(Settlement.SettlementId);
                        }
                    }
                }
            }
            
            if (DeletePayable)
            {
                await _appContext.Db.DeleteNav(payable)
                    .Include(c => c.tb_FM_ReceivablePayableDetails)
                    .ExecuteCommandAsync();
                
                await _appContext.Db.Deleteable<tb_FM_PaymentSettlement>()
                    .In(settlementIdsToDelete.ToArray())
                    .ExecuteCommandAsync();
            }
            
            if (UseTransaction)
            {
                _unitOfWorkManage.CommitTran();
            }
            
            result = true;
        }
        
        return result;
    }
    catch (Exception ex)
    {
        if (UseTransaction)
        {
            _unitOfWorkManage.RollbackTran();
        }
        _logger.Error(ex, $"反核销应收应付单失败: {ex.Message}");
        return false;
    }
}
```

## 8. 关键改进点总结

### 必须实现的检查

1. **状态检查**
   - 检查业务单据状态是否为已确认
   - 检查财务单据状态（草稿/待审核/已审核）
   - 检查状态转换是否允许

2. **核销状态检查**
   - 检查应收/应付款是否已核销
   - 如果已全额核销，不允许反审
   - 如果部分核销，判断是否允许反审

3. **收付款单检查**
   - 检查是否存在已审核的收款单/付款单
   - 如果存在已支付的收付款单，不允许反审

4. **后续单据检查**
   - 检查是否有依赖的单据
   - 如果有，不允许反审

### 必须实现的补偿机制

1. **删除财务单据**
   - 删除应收/应付款单
   - 删除对应的明细

2. **删除核销记录**
   - 删除PaymentSettlement记录
   - 标记或删除相关的核销记录

3. **恢复预收付余额**
   - 如果核销来自预收付，恢复预收付的余额
   - 更新预收付的状态

4. **回写业务单据**
   - 恢复库存
   - 回写采购/销售订单的数量
   - 更新单据状态

5. **审计日志**
   - 记录反审核操作的详细信息
   - 记录财务数据变动
